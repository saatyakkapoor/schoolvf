class UltraAccuratePlateDetector :
    """Multi-algorithm license plate detection for maximum accuracy"""

    def __init__ (self ):

        self .backup_model =None 
        self .model_ready =False 
        self .detection_cache ={}
        self .load_detection_models ()


        self .morphology_kernel =cv2 .getStructuringElement (cv2 .MORPH_RECT ,(3 ,3 ))

        try :

                    print (" Loaded primary license_plate_detector.pt")

                try :

                    self .backup_model .fuse ()
                    print (" Loaded backup YOLOv8n model")
                except :
                    pass

        except Exception as e :

    def detect_plates_ultra_accurate (self ,frame ):
        """Ultra-accurate multi-algorithm plate detection"""
        all_detections =[]


        if self .primary_model :
            yolo_detections =self ._yolo_detection (frame ,self .primary_model ,"primary")
            all_detections .extend (yolo_detections )


        if self .backup_model and len (all_detections )<3 :
            backup_detections =self ._yolo_detection (frame ,self .backup_model ,"backup")
            all_detections .extend (backup_detections )


        cv_detections =self ._enhanced_opencv_detection (frame )
        all_detections .extend (cv_detections )


        merged_detections =self ._merge_detections (all_detections )
        validated_detections =self ._validate_detections (frame ,merged_detections )

        return validated_detections [:Config .MAX_PLATES_PER_FRAME ]

    def _yolo_detection (self ,frame ,model ,model_type ):
        """YOLO-based detection with enhanced post-processing"""
        try :
            results =model (frame ,
            imgsz =Config .AI_INPUT_SIZE ,
            conf =Config .CONFIDENCE_THRESHOLD ,
            iou =Config .NMS_THRESHOLD ,
            verbose =False )

            detections =[]
            if len (results )>0 and hasattr (results [0 ],'boxes')and results [0 ].boxes is not None :
                boxes =results [0 ].boxes 

                for box in boxes :
                    x1 ,y1 ,x2 ,y2 =box .xyxy [0 ].cpu ().numpy ()
                    confidence =box .conf [0 ].cpu ().numpy ()


                    if self ._validate_bbox_geometry (x1 ,y1 ,x2 ,y2 ,frame .shape ):
                        roi =frame [int (y1 ):int (y2 ),int (x1 ):int (x2 )]
                        enhanced_roi =self ._enhance_roi_for_ocr (roi )

                        detections .append ({
                        'bbox':(int (x1 ),int (y1 ),int (x2 ),int (y2 )),
                        'confidence':float (confidence ),
                        'roi':enhanced_roi ,
                        'method':f'yolo_{model_type }',
                        'quality_score':self ._calculate_roi_quality (enhanced_roi )
                        })

            return detections 

        except Exception as e :
            return []

    def _enhanced_opencv_detection (self ,frame ):
        """Enhanced OpenCV detection with multiple filters"""
        detections =[]


        gray =cv2 .cvtColor (frame ,cv2 .COLOR_BGR2GRAY )


        edges =cv2 .Canny (gray ,30 ,200 )
        edges =cv2 .dilate (edges ,self .morphology_kernel ,iterations =1 )
        edges =cv2 .erode (edges ,self .morphology_kernel ,iterations =1 )

        contours ,_ =cv2 .findContours (edges ,cv2 .RETR_EXTERNAL ,cv2 .CHAIN_APPROX_SIMPLE )

        for contour in contours :
            x ,y ,w ,h =cv2 .boundingRect (contour )

            if self ._is_plate_like_region (w ,h ,cv2 .contourArea (contour )):
                roi =frame [y :y +h ,x :x +w ]
                enhanced_roi =self ._enhance_roi_for_ocr (roi )
                quality =self ._calculate_roi_quality (enhanced_roi )

                if quality >0.3 :
                    detections .append ({
                    'bbox':(x ,y ,x +w ,y +h ),
                    'confidence':min (0.8 ,quality ),
                    'roi':enhanced_roi ,
                    'method':'opencv_edge',
                    'quality_score':quality 
                    })

        return detections 

    def _enhance_roi_for_ocr (self ,roi ):
        """Advanced ROI enhancement for better OCR accuracy"""
        if roi is None or roi .size ==0 :
            return None 

        try :

            if len (roi .shape )==3 :
                gray =cv2 .cvtColor (roi ,cv2 .COLOR_BGR2GRAY )
            else :
                gray =roi .copy ()


            height ,width =gray .shape 
            if width <300 :
                scale =300 /width 
                new_width =int (width *scale )
                new_height =int (height *scale )
                gray =cv2 .resize (gray ,(new_width ,new_height ),interpolation =cv2 .INTER_CUBIC )




            enhanced =cv2 .GaussianBlur (gray ,(3 ,3 ),0 )


            clahe =cv2 .createCLAHE (clipLimit =3.0 ,tileGridSize =(8 ,8 ))
            enhanced =clahe .apply (enhanced )


            enhanced =cv2 .bilateralFilter (enhanced ,9 ,75 ,75 )


            kernel =cv2 .getStructuringElement (cv2 .MORPH_RECT ,(2 ,2 ))
            enhanced =cv2 .morphologyEx (enhanced ,cv2 .MORPH_CLOSE ,kernel )


            _ ,enhanced =cv2 .threshold (enhanced ,0 ,255 ,cv2 .THRESH_BINARY +cv2 .THRESH_OTSU )

            return enhanced 

        except Exception as e :
            return roi 

    def _calculate_roi_quality (self ,roi ):
        """Calculate ROI quality score for filtering"""
        if roi is None or roi .size ==0 :
            return 0.0 

        try :

            if len (roi .shape )==3 :
                gray =cv2 .cvtColor (roi ,cv2 .COLOR_BGR2GRAY )
            else :
                gray =roi 


            height ,width =gray .shape 


            size_score =min (1.0 ,(width *height )/10000 )


            contrast =gray .std ()
            contrast_score =min (1.0 ,contrast /50 )


            edges =cv2 .Canny (gray ,50 ,150 )
            edge_density =np .sum (edges >0 )/(width *height )
            edge_score =min (1.0 ,edge_density *10 )


            aspect_ratio =width /height if height >0 else 0 
            if 2.0 <=aspect_ratio <=5.0 :
                aspect_score =1.0 
            else :
                aspect_score =0.5 


            quality =(size_score *0.3 +contrast_score *0.3 +
            edge_score *0.3 +aspect_score *0.1 )

            return quality 

        except :
            return 0.0 

    def _validate_bbox_geometry (self ,x1 ,y1 ,x2 ,y2 ,frame_shape ):
        """Validate bounding box geometry"""
        height ,width =frame_shape [:2 ]


        if x1 <0 or y1 <0 or x2 >=width or y2 >=height :
            return False 


        w ,h =x2 -x1 ,y2 -y1 
        if w <40 or h <15 :
            return False 


        if w >width *0.8 or h >height *0.8 :
            return False 


        aspect_ratio =w /h if h >0 else 0 
        if not (1.5 <=aspect_ratio <=8.0 ):
            return False 

        return True 

    def _is_plate_like_region (self ,width ,height ,area ):
        """Check if region has plate-like characteristics"""
        if height ==0 :
            return False 

        aspect_ratio =width /height 


        if width <60 or height <20 :
            return False 


        if not (2.0 <=aspect_ratio <=6.0 ):
            return False 


        bbox_area =width *height 
        if bbox_area >0 and area /bbox_area <0.3 :
            return False 

        return True 

    def _merge_detections (self ,all_detections ):
        """Merge overlapping detections from different methods"""
        if not all_detections :
            return []


        all_detections .sort (key =lambda x :x ['confidence'],reverse =True )

        merged =[]

        for detection in all_detections :
            is_duplicate =False 

            for existing in merged :
                if self ._calculate_iou (detection ['bbox'],existing ['bbox'])>0.5 :

                    if detection ['confidence']>existing ['confidence']:
                        merged .remove (existing )
                        merged .append (detection )
                    is_duplicate =True 
                    break 

            if not is_duplicate :
                merged .append (detection )

        return merged 

    def _calculate_iou (self ,bbox1 ,bbox2 ):
        """Calculate Intersection over Union"""
        x1_1 ,y1_1 ,x2_1 ,y2_1 =bbox1 
        x1_2 ,y1_2 ,x2_2 ,y2_2 =bbox2 


        x1_i =max (x1_1 ,x1_2 )
        y1_i =max (y1_1 ,y1_2 )
        x2_i =min (x2_1 ,x2_2 )
        y2_i =min (y2_1 ,y2_2 )

        if x2_i <=x1_i or y2_i <=y1_i :
            return 0.0 

        intersection =(x2_i -x1_i )*(y2_i -y1_i )


        area1 =(x2_1 -x1_1 )*(y2_1 -y1_1 )
        area2 =(x2_2 -x1_2 )*(y2_2 -y1_2 )
        union =area1 +area2 -intersection 

        return intersection /union if union >0 else 0.0 

    def _validate_detections (self ,frame ,detections ):
        """Final validation of detections"""
        validated =[]

        for detection in detections :

            if detection ['quality_score']<0.25 :
                continue 


            x1 ,y1 ,x2 ,y2 =detection ['bbox']
            w ,h =x2 -x1 ,y2 -y1 

            if w <50 or h <18 :
                continue 


            validated .append (detection )

        return validated 





class PerformanceMonitor :
    """Real-time performance monitoring and adaptive optimization"""

    def __init__ (self ):
        self .fps_history =deque (maxlen =30 )
        self .processing_times =deque (maxlen =50 )
        self .last_fps_update =time .time ()
        self .current_fps =0 
        self .target_fps =Config .TARGET_FPS 

    def update_fps (self ,fps ):
        """Update FPS and trigger adaptive adjustments"""
        self .fps_history .append (fps )
        self .current_fps =fps 

        if len (self .fps_history )>=10 :
            avg_fps =sum (self .fps_history )/len (self .fps_history )
            return self ._get_performance_mode (avg_fps )
        return "NORMAL"

    def _get_performance_mode (self ,avg_fps ):
        """Determine performance mode based on FPS"""
        if avg_fps <Config .FPS_TARGET_MIN :
            return "SPEED_BOOST"
        elif avg_fps >Config .FPS_TARGET_MAX :
            return "QUALITY_BOOST"
        else :
            return "NORMAL"

    def get_adaptive_settings (self ):
        """Get adaptive settings based on performance"""
        mode =self ._get_performance_mode (self .current_fps )

            'engine_count':1 ,
            'thread_count':2

            return {

    def init_adaptive_ocr (self ):
        """Initialize ADAPTIVE OCR system with smart loading"""
        print (" Initializing ADAPTIVE OCR engine...")


        try :

            engine =easyocr .Reader (['en'],gpu =True )
            self .ocr_engines .append (('EasyOCR_Primary',engine ))
            print (" Primary EasyOCR engine loaded (GPU)")
        except :
            try :
                engine =easyocr .Reader (['en'],gpu =True )
                self .ocr_engines .append (('EasyOCR_Primary',engine ))
                print (" Primary EasyOCR engine loaded (CPU)")
            except :
                print (" Failed to load OCR engine")
                return 


        if Config .PERFORMANCE_MODE !="SPEED":
            try :
                backup_engine =easyocr .Reader (['en'],gpu =True )
                self .ocr_engines .append (('EasyOCR_Backup',backup_engine ))
                print (" Backup EasyOCR engine loaded")
            except :
                pass 

        if self .ocr_engines :
            self .ocr_ready =True 
            print (f"🎯 ADAPTIVE OCR system ready with {len (self .ocr_engines )} engines!")
        else :
            print (" No OCR engines available")

    def start_adaptive_threads (self ):
        """Start ADAPTIVE OCR worker threads"""
        self .running =True 


        for i in range (self .current_thread_count ):
            thread =threading .Thread (target =self ._adaptive_ocr_worker ,daemon =True )
            thread .start ()
            self .processing_threads .append (thread )

        print (f"🚀 Started {self .current_thread_count } adaptive OCR workers")

    def process_plate_async (self ,plate_data ,frame_id ,camera_id ):
        """Add plate for enhanced async OCR processing"""
        if not self .ocr_ready :
            return 

        try :
            self .ocr_queue .put_nowait ({
            'roi':plate_data ['roi'],
            'bbox':plate_data ['bbox'],
            'confidence':plate_data ['confidence'],
            'method':plate_data ['method'],
            'quality_score':plate_data ['quality_score'],
            'frame_id':frame_id ,
            'camera_id':camera_id ,
            'timestamp':time .time ()
            })
        except queue .Full :
            pass 

    def get_ocr_results (self ):
        """Get all available enhanced OCR results"""
        results =[]

        try :
            while True :
                result =self .result_queue .get_nowait ()
                results .append (result )
        except queue .Empty :
            pass 

        return results 

    def _adaptive_ocr_worker (self ):
        """ADAPTIVE OCR worker with smart processing"""
        while self .running :
            try :
                plate_data =self .ocr_queue .get (timeout =0.02 )


                settings =self .performance_monitor .get_adaptive_settings ()


                engine_to_use =self .ocr_engines [0 ]if self .ocr_engines else None 
                if not engine_to_use :
                    continue 

                engine_name ,engine =engine_to_use 

                try :

                    text_result =self ._extract_text_adaptive (
                    plate_data ['roi'],
                    engine ,
                    engine_name ,
                    settings 
                    )

                    if text_result :

                        final_result ={
                        'text':text_result ['text'],
                        'bbox':plate_data ['bbox'],
                        'confidence':plate_data ['confidence'],
                        'ocr_confidence':text_result ['confidence'],
                        'final_confidence':text_result ['confidence'],
                        'method':f"adaptive_{engine_name }",
                        'quality_score':plate_data ['quality_score'],
                        'frame_id':plate_data ['frame_id'],
                        'camera_id':plate_data ['camera_id'],
                        'timestamp':plate_data ['timestamp'],
                        'is_emergency':self ._is_emergency_plate (text_result ['text']),
                        'validation_score':self ._fast_validation (text_result ['text'])
                        }

                        try :
                            self .result_queue .put_nowait (final_result )
                        except queue .Full :
                            pass 

                except Exception as e :
                    continue 

            except queue .Empty :
                continue 
            except Exception as e :
                continue 

    def _extract_text_adaptive (self ,roi ,engine ,engine_name ,settings ):
        """ADAPTIVE text extraction with smart optimization"""
        if roi is None or roi .size ==0 :
            return None 

        try :

            if not settings .get ('skip_preprocessing',False ):
                roi =self ._smart_preprocess (roi ,settings )


            results =engine .readtext (
            roi ,
            detail =1 ,
            paragraph =False ,
            width_ths =0.7 if settings .get ('fast_mode',True )else 0.5 ,
            height_ths =0.7 if settings .get ('fast_mode',True )else 0.5 
            )


            best_result =None 
            best_confidence =0 

            for bbox ,text ,confidence in results :
                if confidence >Config .OCR_MIN_CONFIDENCE :

    def _fast_text_cleaning (self ,text ):
        """FAST text cleaning optimized for Indian plates"""
        if not text :
            return None 


        text =text .upper ().strip ()
        text =re .sub (r'[^A-Z0-9]','',text )


        char_fixes ={
        'O':'0','0':'O',
        'I':'1','1':'I',
        'S':'5','5':'S',
        'B':'8','8':'B',
        'G':'6','6':'G',
        'Z':'2','2':'Z'
        }


        fixed_text =""
        for i ,char in enumerate (text ):
            if i <2 :
                if char .isdigit ()and char in char_fixes :
                    fixed_text +=char_fixes [char ]if char_fixes [char ].isalpha ()else char 
                else :
                    fixed_text +=char 
            elif 2 <=i <4 :
                if char .isalpha ()and char in char_fixes :
                    fixed_text +=char_fixes [char ]if char_fixes [char ].isdigit ()else char 
                else :
                    fixed_text +=char 
            elif 4 <=i <6 :
                if char .isdigit ()and char in char_fixes :
                    fixed_text +=char_fixes [char ]if char_fixes [char ].isalpha ()else char 
                else :
                    fixed_text +=char 
            else :
                if char .isalpha ()and char in char_fixes :
                    fixed_text +=char_fixes [char ]if char_fixes [char ].isdigit ()else char 
                else :
                    fixed_text +=char 

        return fixed_text if len (fixed_text )>=6 else None 

    def _fast_validation (self ,text ):
        """FAST validation for Indian license plates"""
        if not text or len (text )<6 :
            return 0.0 

        score =0.0 


        indian_patterns =[

        r'^[A-Z]{2}[0-9]{2}[A-Z]{1,2}[0-9]{4}$',

        r'^[A-Z]{2}[0-9]{2}[A-Z]{1}[0-9]{4}$',

        r'^[A-Z]{2}[0-9]{2}[A-Z]{2}[0-9]{3}[A-Z]$',

        r'^(AMB|POL|FIRE|ARMY|NAVY|AF)[0-9A-Z]{4,8}$',

        r'^[A-Z]{2}[0-9]{1,2}[A-Z]{1,3}[0-9]{1,4}$'
        ]


        for pattern in indian_patterns :
            if re .match (pattern ,text ):
                score +=0.8 
                break 
        else :

            if re .match (r'^[A-Z]{2}',text ):
                score +=0.3 
            if re .search (r'[0-9]{4}$',text ):
                score +=0.3 
            if 8 <=len (text )<=10 :
                score +=0.2 


        indian_state_codes ={
        'DL','MH','KA','TN','AP','TS','UP','MP','RJ','GJ',
        'WB','OR','BR','JH','HR','PB','HP','UK','CH','GA',
        'SK','ML','MN','TR','NL','MZ','AR','AS','AN','JK',
        'LA','LD','PY','DN','DD','CG'
        }

        if len (text )>=2 and text [:2 ]in indian_state_codes :
            score +=0.2 

        return min (1.0 ,score )

    def _validate_plate_text (self ,text ):
        """Enhanced plate text validation with scoring"""
        if not text or len (text )<6 :
            return 0.0 

        score =0.0 


        if 8 <=len (text )<=12 :
            score +=0.3 
        elif 6 <=len (text )<8 :
            score +=0.2 
        else :
            score +=0.1 


        patterns =[
        r'^[A-Z]{2}[0-9]{2}[A-Z]{1,2}[0-9]{4}$',
        r'^[A-Z]{2}[0-9]{1,2}[A-Z]{1,3}[0-9]{1,4}$',
        r'^[A-Z]{3}[0-9]{3,4}$',
        r'^[0-9]{2}[A-Z]{2}[0-9]{4}$',
        ]

        for pattern in patterns :
            if re .match (pattern ,text ):
                score +=0.4 
                break 
        else :

            if re .match (r'^[A-Z]{2}',text ):
                score +=0.2 
            if re .search (r'[0-9]{3,4}$',text ):
                score +=0.2 


        letters =sum (1 for c in text if c .isalpha ())
        numbers =sum (1 for c in text if c .isdigit ())

        if 2 <=letters <=6 and numbers >=3 :
            score +=0.3 

        return min (1.0 ,score )

    def _is_emergency_plate (self ,text ):
        """Enhanced emergency vehicle detection with file-based registry"""
        if not text :
            return False 


        if hasattr (self ,'emergency_registry'):
            if text in self .emergency_registry :
                return True 


        emergency_patterns =[
        r'^AMB[0-9A-Z]*',
        r'^POL[0-9A-Z]*',
        r'^FIRE[0-9A-Z]*',

class RoadDividerManager :
    """Professional Road Divider and Lane Management System"""

    def __init__ (self ,frame_width ,frame_height ):
        self .frame_width =frame_width 
        self .frame_height =frame_height 
        self .lanes =[]
        self .divider_lines =[]
        self .lane_boundaries =[]
        self .setup_road_layout ()


        self .vehicle_tracks ={}
        self .violation_history ={}

    def setup_road_layout (self ):
        """Setup professional road layout with lanes and dividers"""
        if not Config .ENABLE_ROAD_DIVIDERS :
            return 


        usable_width =int (self .frame_width *0.8 )
        start_x =(self .frame_width -usable_width )//2 
        lane_width =usable_width //Config .NUM_LANES 


        for i in range (Config .NUM_LANES +1 ):
            x_pos =start_x +(i *lane_width )
            self .lane_boundaries .append (x_pos )


        center_divider_x =start_x +(usable_width //2 )
        self .divider_lines .append ({
        'x':center_divider_x ,
        'type':'center_divider',
        'color':Config .DIVIDER_COLOR ,
        'thickness':Config .DIVIDER_THICKNESS 
        })


        for i in range (1 ,Config .NUM_LANES ):
            if i !=Config .NUM_LANES //2 :
                x_pos =start_x +(i *lane_width )
                self .divider_lines .append ({
                'x':x_pos ,
                'type':'lane_marker',
                'color':Config .LANE_LINE_COLOR ,
                'thickness':Config .LANE_LINE_THICKNESS 
                })


        for i in range (Config .NUM_LANES ):
            left_boundary =start_x +(i *lane_width )
            right_boundary =start_x +((i +1 )*lane_width )
            direction ='left_to_right'if i <Config .NUM_LANES //2 else 'right_to_left'

            self .lanes .append ({
            'id':i ,
            'left_boundary':left_boundary ,
            'right_boundary':right_boundary ,
            'center_x':(left_boundary +right_boundary )//2 ,
            'direction':direction ,
            'violations':0 
            })

    def draw_road_system (self ,frame ):
        """Draw professional road dividers and lane markers"""
        if not Config .ENABLE_ROAD_DIVIDERS :
            return frame 


        overlay =frame .copy ()


        for divider in self .divider_lines :
            if divider ['type']=='center_divider':
                cv2 .line (overlay ,
                (divider ['x'],0 ),
                (divider ['x'],self .frame_height ),
                self ._hex_to_bgr (divider ['color']),
                divider ['thickness'])
            else :
                self ._draw_dashed_line (overlay ,divider ['x'],
                self ._hex_to_bgr (divider ['color']),
                divider ['thickness'])


        frame =cv2 .addWeighted (frame ,0.85 ,overlay ,0.15 ,0 )
        return frame 

    def _draw_dashed_line (self ,frame ,x ,color ,thickness ):
        dash_length =40 
        gap_length =16 
        y =0 

        while y <self .frame_height :
            cv2 .line (frame ,(x ,y ),(x ,min (y +dash_length ,self .frame_height )),
            color ,thickness )
            y +=dash_length +gap_length 

    def _hex_to_bgr (self ,hex_color ):
        """Convert hex color to BGR tuple"""
        hex_color =hex_color .lstrip ('#')
        return tuple (int (hex_color [i :i +2 ],16 )for i in (4 ,2 ,0 ))

    def detect_lane_violation (self ,vehicle_bbox ,vehicle_id ,timestamp ):
        """Detect if vehicle crosses lane dividers (rash driving)"""
        if not Config .RASH_DRIVING_ENABLED :
            return None 

        x1 ,y1 ,x2 ,y2 =vehicle_bbox 
        vehicle_center_x =(x1 +x2 )//2 
        vehicle_center_y =(y1 +y2 )//2 


        current_lane =self ._get_vehicle_lane (vehicle_center_x )
        if current_lane is None :
            return None 


        if vehicle_id not in self .vehicle_tracks :
            self .vehicle_tracks [vehicle_id ]={
            'positions':[],
            'lanes':[],
            'violations':[],
            'last_violation_time':0 
            }

        track =self .vehicle_tracks [vehicle_id ]
        track ['positions'].append ((vehicle_center_x ,vehicle_center_y ,timestamp ))
        track ['lanes'].append (current_lane )


        if len (track ['positions'])>30 :
            track ['positions']=track ['positions'][-30 :]
            track ['lanes']=track ['lanes'][-30 :]


        violation =self ._analyze_lane_behavior (vehicle_id ,vehicle_bbox ,timestamp )

        if violation :
            track ['violations'].append (violation )
            track ['last_violation_time']=timestamp 


            self ._log_lane_violation (vehicle_id ,violation ,timestamp )

        return violation 

    def _get_vehicle_lane (self ,vehicle_center_x ):
        """Determine which lane the vehicle is in"""
        for lane in self .lanes :
            if lane ['left_boundary']<=vehicle_center_x <=lane ['right_boundary']:
                return lane ['id']
        return None 

    def _analyze_lane_behavior (self ,vehicle_id ,vehicle_bbox ,timestamp ):
        """Analyze vehicle behavior for lane violations"""
        track =self .vehicle_tracks [vehicle_id ]

        if len (track ['positions'])<5 :
            return None 


        violation =self ._check_center_divider_crossing (vehicle_id ,vehicle_bbox ,timestamp )
        if violation :
            return violation 


        violation =self ._check_rapid_lane_changes (vehicle_id ,timestamp )
        if violation :
            return violation 


        violation =self ._check_erratic_movement (vehicle_id ,timestamp )
        if violation :
            return violation 

        return None 

    def _check_center_divider_crossing (self ,vehicle_id ,vehicle_bbox ,timestamp ):
        """Check if vehicle crosses the center divider"""
        x1 ,y1 ,x2 ,y2 =vehicle_bbox 
        track =self .vehicle_tracks [vehicle_id ]


        center_divider =None 
        for divider in self .divider_lines :
            if divider ['type']=='center_divider':
                center_divider =divider ['x']
                break 

        if center_divider is None :
            return None 


        buffer =Config .LANE_VIOLATION_BUFFER 
        if (x1 -buffer )<=center_divider <=(x2 +buffer ):

            if len (track ['lanes'])>=3 :
                recent_lanes =track ['lanes'][-3 :]
                left_side_lanes =[i for i in recent_lanes if i <Config .NUM_LANES //2 ]
                right_side_lanes =[i for i in recent_lanes if i >=Config .NUM_LANES //2 ]

                if left_side_lanes and right_side_lanes :
                    return {
                    'type':'CENTER_DIVIDER_CROSSING',
                    'severity':'HIGH',
                    'description':'Vehicle crossed center divider - DANGEROUS!',
                    'position':(x1 ,y1 ,x2 ,y2 ),
                    'timestamp':timestamp ,
                    'lane_from':min (recent_lanes ),
                    'lane_to':max (recent_lanes )
                    }
        return None 

    def _check_rapid_lane_changes (self ,vehicle_id ,timestamp ):
        """Detect rapid/unsafe lane changes"""
        track =self .vehicle_tracks [vehicle_id ]

        if len (track ['lanes'])<10 :
            return None 

        recent_lanes =track ['lanes'][-10 :]
        recent_times =[pos [2 ]for pos in track ['positions'][-10 :]]


        lane_changes =0 
        for i in range (1 ,len (recent_lanes )):
            if recent_lanes [i ]!=recent_lanes [i -1 ]:
                lane_changes +=1 


        time_span =recent_times [-1 ]-recent_times [0 ]
        if time_span >0 and lane_changes >=3 and time_span <5.0 :
            return {
            'type':'RAPID_LANE_CHANGES',
            'severity':'MEDIUM',
            'description':f'{lane_changes } lane changes in {time_span :.1f}s - Rash driving!',
            'timestamp':timestamp ,
            'lane_changes':lane_changes ,
            'time_span':time_span 
            }
        return None 

    def _check_erratic_movement (self ,vehicle_id ,timestamp ):
        """Detect erratic/zigzag movement patterns"""
        track =self .vehicle_tracks [vehicle_id ]

        if len (track ['positions'])<8 :
            return None 

        positions =track ['positions'][-8 :]


        direction_changes =0 
        prev_direction =None 

        for i in range (1 ,len (positions )):
            dx =positions [i ][0 ]-positions [i -1 ][0 ]
            if abs (dx )>5 :
                current_direction ='left'if dx <0 else 'right'
                if prev_direction and prev_direction !=current_direction :
                    direction_changes +=1 
                prev_direction =current_direction 


        if direction_changes >=4 :
            return {
            'type':'ERRATIC_MOVEMENT',
            'severity':'MEDIUM',
            'description':'Erratic zigzag movement detected - Possible rash driving!',
            'timestamp':timestamp ,
            'direction_changes':direction_changes 
            }
        return None 

    def _log_lane_violation (self ,vehicle_id ,violation ,timestamp ):
        """Log lane violation to file"""
        try :
            os .makedirs ("rash_logs",exist_ok =True )

            log_entry ={
            'timestamp':datetime .now ().isoformat (),
            'vehicle_id':vehicle_id ,
            'violation_type':violation ['type'],
            'severity':violation ['severity'],
            'description':violation ['description'],
            'details':violation 
            }

            log_file =f"rash_logs/lane_violations_{datetime .now ().strftime ('%Y%m%d')}.json"


            logs =[]
            if os .path .exists (log_file ):
                try :
                    with open (log_file ,'r')as f :
                        logs =json .load (f )
                except :
                    logs =[]

            logs .append (log_entry )


            with open (log_file ,'w')as f :
                json .dump (logs ,f ,indent =2 )

            print (f"🚨 LANE VIOLATION LOGGED: {violation ['type']} - {violation ['description']}")

        except Exception as e :
            print (f"Error logging lane violation: {e }")

    def get_violation_stats (self ):
        """Get lane violation statistics"""
        total_violations =0 
        violation_types ={}

        for track in self .vehicle_tracks .values ():
            total_violations +=len (track ['violations'])
            for violation in track ['violations']:
                vtype =violation ['type']
                violation_types [vtype ]=violation_types .get (vtype ,0 )+1 

        return {
        'total_violations':total_violations ,
        'violation_types':violation_types ,
        'active_tracks':len (self .vehicle_tracks )
        }

class SmartTrafficDetector :
    """Enhanced AI-powered traffic violation and emergency detection"""
    def __init__ (self ):
        self .violation_memory ={}
        self .emergency_memory ={}
        self .saved_violations =set ()
        self .plate_history ={}

    def detect_emergency_vehicle (self ,frame ,detections ):
        """Detect emergency vehicles using AI and visual cues"""
        emergency_vehicles =[]

        for detection in detections :
            x1 ,y1 ,x2 ,y2 ,conf ,cls =detection 
            if conf >0.5 :

                roi =frame [int (y1 ):int (y2 ),int (x1 ):int (x2 )]


                if self ._is_emergency_vehicle (roi ,cls ):
                    emergency_vehicles .append ({
                    'bbox':(x1 ,y1 ,x2 ,y2 ),
                    'confidence':conf ,
                    'type':'emergency'
                    })

        return emergency_vehicles 

    def _is_emergency_vehicle (self ,roi ,vehicle_class ):
        """Analyze ROI to determine if it's an emergency vehicle"""
        if roi .size ==0 :
            return False 


        hsv =cv2 .cvtColor (roi ,cv2 .COLOR_BGR2HSV )


        red_mask1 =cv2 .inRange (hsv ,np .array ([0 ,50 ,50 ]),np .array ([10 ,255 ,255 ]))
        red_mask2 =cv2 .inRange (hsv ,np .array ([170 ,50 ,50 ]),np .array ([180 ,255 ,255 ]))
        blue_mask =cv2 .inRange (hsv ,np .array ([100 ,50 ,50 ]),np .array ([130 ,255 ,255 ]))

        red_pixels =cv2 .countNonZero (red_mask1 )+cv2 .countNonZero (red_mask2 )
        blue_pixels =cv2 .countNonZero (blue_mask )

        total_pixels =roi .shape [0 ]*roi .shape [1 ]
        emergency_ratio =(red_pixels +blue_pixels )/total_pixels 


        return emergency_ratio >0.05 or vehicle_class ==AMBULANCE_CLASS_ID 

    def detect_violations (self ,frame ,detections ,traffic_state ):
        """Detect traffic violations including red light running"""
        violations =[]

        for detection in detections :
            x1 ,y1 ,x2 ,y2 ,conf ,cls =detection 
            if conf >0.5 :
                center_x =(x1 +x2 )/2 
                center_y =(y1 +y2 )/2 


                if traffic_state =='red'and self ._is_in_intersection (center_x ,center_y ,frame .shape ):
                    violations .append ({
                    'type':'red_light_violation',
                    'bbox':(x1 ,y1 ,x2 ,y2 ),
                    'position':(center_x ,center_y ),
                    'confidence':conf 
                    })


                if self ._detect_rash_driving (center_x ,center_y ,frame .shape ):
                    violations .append ({
                    'type':'rash_driving',
                    'bbox':(x1 ,y1 ,x2 ,y2 ),
                    'position':(center_x ,center_y ),
                    'confidence':conf 
                    })

        return violations 

    def _is_in_intersection (self ,x ,y ,frame_shape ):
        """Check if vehicle is in intersection area"""
        h ,w =frame_shape [:2 ]
        intersection_zone ={
        'x_min':w *0.3 ,
        'x_max':w *0.7 ,
        'y_min':h *0.3 ,
        'y_max':h *0.7 
        }
        return (intersection_zone ['x_min']<x <intersection_zone ['x_max']and 
        intersection_zone ['y_min']<y <intersection_zone ['y_max'])

    def _detect_rash_driving (self ,x ,y ,frame_shape ):
        """Enhanced rash driving detection with lane analysis"""




        if hasattr (self ,'prev_positions'):
            if (x ,y )in self .prev_positions :

                movements =[abs (px -x )+abs (py -y )for px ,py in self .prev_positions [-5 :]]
                if len (movements )>3 :
                    avg_movement =sum (movements )/len (movements )
                    if avg_movement >Config .ERRATIC_MOVEMENT_THRESHOLD :
                        return True 
        else :
            self .prev_positions =[]

        self .prev_positions .append ((x ,y ))
        if len (self .prev_positions )>10 :
            self .prev_positions =self .prev_positions [-10 :]

        return False 

class EnhancedPlateProcessor :
    """Improved license plate detection and recognition"""
    def __init__ (self ):
        self .prev_plates ={}
        self .emergency_plates =set ()

    def process_plate (self ,plate_roi ):
        """Enhanced plate processing with better accuracy"""
        if plate_roi .size ==0 :
            return None 


        enhanced =self ._enhance_plate_image (plate_roi )
        text =self ._ocr_plate (enhanced )

        if text and self ._validate_plate (text ):
            return text 
        return None 

    def _enhance_plate_image (self ,plate_img ):
        """Advanced plate image enhancement"""
        gray =cv2 .cvtColor (plate_img ,cv2 .COLOR_BGR2GRAY )


        clahe =cv2 .createCLAHE (clipLimit =3.0 ,tileGridSize =(8 ,8 ))
        enhanced =clahe .apply (gray )


        denoised =cv2 .fastNlMeansDenoising (enhanced )


        _ ,thresh =cv2 .threshold (denoised ,0 ,255 ,cv2 .THRESH_BINARY +cv2 .THRESH_OTSU )

        return thresh 

    def _ocr_plate (self ,processed_plate ):
        """OCR processing with error correction"""
        if reader is None :
            return None 

        try :
            results =reader .readtext (processed_plate ,detail =0 ,paragraph =False )
            if results :
                text =''.join (results ).upper ()

                clean_text =re .sub (r'[^A-Z0-9]','',text )
                return clean_text 
        except :
            return None 
        return None 

    def _validate_plate (self ,text ):
        """Validate Indian license plate format"""
        if not text or len (text )<8 :
            return False 


        patterns =[
        r'^[A-Z]{2}[0-9]{2}[A-Z]{1,2}[0-9]{4}$',
        r'^[A-Z]{2}[0-9]{1,2}[A-Z]{1,3}[0-9]{1,4}$'
        ]

        return any (re .match (pattern ,text )for pattern in patterns )

class TrafficLightController :
    """Smart traffic light controller with emergency prioritization (EAST/WEST only)"""
    def __init__ (self ,serial_port =None ):

        self .state ={'east':'red','west':'red'}
        self .normal_cycle =['east','west']
        self .current_phase =0 
        self .green_duration =10 
        self .yellow_duration =3 
        self .timer =self .green_duration 
        self .phase_state ='green'
        self .emergency_override =False 
        self .emergency_direction =None 
        self .serial_port =serial_port 
        self .state_changed_callback =None 

    def update_cycle (self ):
        """Update traffic light cycle"""
        if self .emergency_override :
            return self ._handle_emergency_cycle ()

        return self ._handle_normal_cycle ()

    def _handle_normal_cycle (self ):
        """Handle normal traffic cycle"""
        active_direction =self .normal_cycle [self .current_phase ]
        self .timer -=1 

        if self .timer <0 :
            if self .phase_state =='green':
                self .phase_state ='yellow'
                self .timer =self .yellow_duration 
            else :
                self .phase_state ='green'
                self .current_phase =(self .current_phase +1 )%len (self .normal_cycle )
                self .timer =self .green_duration 


        for direction in self .state :
            if direction ==active_direction :
                self .state [direction ]=self .phase_state 
            else :
                self .state [direction ]='red'

        self ._send_serial_signal (active_direction ,self .phase_state )
        return self .state 

    def _handle_emergency_cycle (self ):
        """Handle emergency vehicle prioritization (EAST/WEST only)"""
        if self .emergency_direction in self .state :
            for direction in self .state :
                self .state [direction ]='green'if direction ==self .emergency_direction else 'red'
            self ._send_serial_signal (self .emergency_direction ,'green')
        return self .state 

    def trigger_emergency (self ,direction ):
        """Trigger emergency vehicle priority"""
        self .emergency_override =True 
        self .emergency_direction =direction 
        print (f"EMERGENCY: Prioritizing {direction .upper ()} direction")
        return self .state 

    def clear_emergency (self ):
        """Clear emergency override"""
        self .emergency_override =False 
        self .emergency_direction =None 
        print ("Emergency override cleared - resuming normal cycle")
        return self .state 

    def _send_serial_signal (self ,direction ,state ):
        """Send serial signals for traffic light changes"""
        if state =='green'and self .serial_port :
            signal_map ={
            'east':'EG',
            'west':'WG'
            }

            signal =signal_map .get (direction )
            if signal :
                try :
                    self .serial_port .write (f"{signal }\n".encode ())
                    print (f"Sent serial signal: {signal }")
                except Exception as e :
                    print (f"Serial send error: {e }")

class SerialManager :
    """Manages serial communication with auto-detection"""
    def __init__ (self ):

    def connect (self ):
        """Auto-detect and connect to serial port"""
        ports_to_try =[]

        elif platform .system ()=="Windows":
            ports_to_try =[f"COM{i }"for i in range (1 ,20 )]
        else :
            ports_to_try =[f"/dev/ttyUSB{i }"for i in range (10 )]+[f"/dev/ttyACM{i }"for i in range (10 )]

        for port in ports_to_try :
            try :
                self .serial_port =serial .Serial (port ,115200 ,timeout =1 )
                print (f"Serial connected to {port }")
                return 
            except :
                continue 

        print ("No serial port found")





class SerialReaderThread (QThread ):
    """Background serial reader emitting lines to UI"""
    line_received =pyqtSignal (str )
    status_changed =pyqtSignal (str )
    error_occurred =pyqtSignal (str )

    def __init__ (self ,port_name :str ,baud_rate :int =115200 ,parent =None ):
        super ().__init__ (parent )
        self .port_name =port_name 
        self .baud_rate =baud_rate 
        self ._running =False 
        self ._ser =None 

    def run (self ):
        try :
            self ._ser =serial .Serial (self .port_name ,self .baud_rate ,timeout =1 )
            self .status_changed .emit (f"Connected {self .port_name } @ {self .baud_rate }")
            self ._running =True 
            while self ._running :
                try :
                    raw =self ._ser .readline ()
                    if not raw :
                        continue 
                    try :
                        line =raw .decode (errors ='ignore').strip ()
                    except Exception :
                        line =str (raw )
                    if line :
                        self .line_received .emit (line )
                except Exception as e :
                    self .error_occurred .emit (str (e ))
                    break 
        except Exception as e :
            self .error_occurred .emit (f"Serial open failed: {e }")
        finally :
            try :
                if self ._ser and self ._ser .is_open :
                    self ._ser .close ()
            except Exception :
                pass 
            self .status_changed .emit ("Disconnected")

    def stop (self ):
        self ._running =False 
        try :
            if self ._ser and self ._ser .is_open :
                self ._ser .close ()
        except Exception :
            pass

        self .emergency_duration =30 

    def trigger_emergency (self ,direction ,plate_number ):
        """Auto-trigger emergency for detected emergency vehicle"""
        direction_map ={
        0 :'north',
        1 :'east',
        2 :'south',
        3 :'west'
        }

        emergency_dir =direction_map .get (direction ,'north')

        if not self .emergency_active :
            self .emergency_active =True 
            self .emergency_direction =emergency_dir 
            self .emergency_start_time =time .time ()


            for direction in self .current_state :
                if direction ==emergency_dir :
                    self .current_state [direction ]='green'
                else :
                    self .current_state [direction ]='red'

            print (f"🚨 EMERGENCY TRIGGERED: {emergency_dir .upper ()} direction for {plate_number }")


            self ._log_emergency_event (emergency_dir ,plate_number )

    def update_cycle (self ):
        """Update traffic light cycle"""
        current_time =time .time ()


        if self .emergency_active :
            if current_time -self .emergency_start_time >self .emergency_duration :
                self .clear_emergency ()
            return self .current_state 



        return self .current_state 

    def clear_emergency (self ):
        """Clear emergency state"""
        self .emergency_active =False 
        self .emergency_direction =None 
        self .emergency_start_time =None 
        print (" Emergency cleared - resuming normal traffic cycle")

    def _log_emergency_event (self ,direction ,plate_number ):
        """Log emergency events to file"""
        try :
            timestamp =datetime .now ().strftime ("%Y-%m-%d %H:%M:%S")
            with open ('emergency_log.txt','a')as f :
                f .write (f"{timestamp } - EMERGENCY: {direction .upper ()} direction triggered by {plate_number }\n")
        except Exception as e :
            print (f"❌ Error logging emergency: {e }")





class HighPerformanceCamera (QThread ):
    """Ultra-accurate camera processing with enhanced detection"""

    frame_ready =pyqtSignal (int ,np .ndarray ,dict )
    plate_detected =pyqtSignal (int ,str ,dict )
    violation_alert =pyqtSignal (dict )

    def __init__ (self ,camera_id ,mode ="PRODUCTION"):
        super ().__init__ ()
        self .camera_id =camera_id 
        self .mode =mode 
        self .running =False 
        self .cap =None 


        self .frame_count =0 
        self .fps_start =time .time ()
        self .fps_counter =0 
        self .current_fps =0 


        self .plate_detector =UltraAccuratePlateDetector ()
        self .ocr_processor =AdaptiveOCRProcessor ()


        self .vehicle_detector =VehicleDetector ()


        self .traffic_controller =SmartTrafficController ()


        self .road_divider_manager =RoadDividerManager (Config .CAMERA_WIDTH ,Config .CAMERA_HEIGHT )


        self .frame_skip_counter =0 
        self .last_frame_time =time .time ()
        self .stable_detections =[]

    def set_mode (self ,mode ):
        """Switch between TEST and PRODUCTION modes"""
        self .mode =mode 

    def run (self ):
        """Main camera processing loop"""
        self .running =True 


        self .cap =cv2 .VideoCapture (self .camera_id )

        if self .cap .isOpened ():

            self .cap .set (cv2 .CAP_PROP_FRAME_WIDTH ,Config .CAMERA_WIDTH )
            self .cap .set (cv2 .CAP_PROP_FRAME_HEIGHT ,Config .CAMERA_HEIGHT )
            self .cap .set (cv2 .CAP_PROP_FPS ,Config .TARGET_FPS )
            self .cap .set (cv2 .CAP_PROP_BUFFERSIZE ,1 )
            print (f"✅ Camera {self .camera_id } initialized with enhanced settings")
        else :
            print (f"❌ Camera {self .camera_id } failed to open")
            return 

        while self .running :
            start_time =time .time ()

            ret ,frame =self .cap .read ()
            if not ret :
                continue 

            self .frame_count +=1 
            self .frame_skip_counter +=1 


            frame_data =self ._process_frame_adaptive (frame )


            self .frame_ready .emit (self .camera_id ,frame ,frame_data )


            self ._update_fps ()


            elapsed =time .time ()-start_time 
            target_frame_time =1.0 /Config .TARGET_FPS 
            if elapsed <target_frame_time :
                time .sleep (target_frame_time -elapsed )

    def _process_frame_adaptive (self ,frame ):
        """ADAPTIVE frame processing with smart performance optimization"""
        frame_data ={
        'plates':[],
        'detected_texts':[],
        'fps':self .current_fps ,
        'mode':self .mode ,
        'processing':False ,
        'accuracy_score':0.0 ,
        'performance_mode':'NORMAL'
        }


        performance_mode =self .ocr_processor .performance_monitor .update_fps (self .current_fps )
        frame_data ['performance_mode']=performance_mode 


        if self .frame_skip_counter >=Config .PROCESS_EVERY_N_FRAMES :
            self .frame_skip_counter =0 
            frame_data ['processing']=True 


            detected_plates =self .plate_detector .detect_plates_ultra_accurate (frame )
            frame_data ['plates']=detected_plates 


            if detected_plates :
                quality_scores =[p ['quality_score']for p in detected_plates ]
                frame_data ['accuracy_score']=sum (quality_scores )/len (quality_scores )


            for plate in detected_plates :
                self .ocr_processor .process_plate_async (
                plate ,self .frame_count ,self .camera_id 
                )


            if Config .RASH_DRIVING_ENABLED and Config .VEHICLE_DETECTION_ENABLED :
                violations =[]
                timestamp =time .time ()


                vehicle_dets =self .vehicle_detector .detect (frame )


                center_x =None 
                if self .road_divider_manager .divider_lines :
                    for div in self .road_divider_manager .divider_lines :
                        if div ['type']=='center_divider':
                            center_x =div ['x']
                            break 

                if center_x is not None :
                    for det in vehicle_dets :
                        x1 ,y1 ,x2 ,y2 ,conf ,cls =det 

                        if (x1 -Config .LANE_VIOLATION_BUFFER )<=center_x <=(x2 +Config .LANE_VIOLATION_BUFFER ):
                            violation ={
                            'type':'CENTER_DIVIDER_CROSSING',
                            'severity':'HIGH',
                            'description':'Vehicle crossed center divider',
                            'position':(x1 ,y1 ,x2 ,y2 ),
                            'timestamp':timestamp ,
                            'vehicle_id':f"c_{x1 }_{y1 }_{int (conf *100 )}",
                            'cls':int (cls ),
                            'conf':float (conf )
                            }
                            violations .append (violation )
                            frame_data .setdefault ('violations',[]).append (violation )

                if violations :
                    frame_data ['lane_violations']=len (violations )
                    frame_data ['violation_stats']=self .road_divider_manager .get_violation_stats ()

                frame_data ['vehicle_detections']=vehicle_dets


        ocr_results =self .ocr_processor .get_ocr_results ()


        indian_results =[]
        for result in ocr_results :
            if (result ['camera_id']==self .camera_id and 
            result .get ('validation_score',0 )>0.8 ):
                indian_results .append (result )


                if result .get ('is_emergency',False ):
                    self .traffic_controller .trigger_emergency (
                    self .camera_id ,
                    result ['text']
                    )


        current_time =time .time ()
        if current_time -self .last_frame_time >0.5 :
            self .stable_detections =indian_results 
            self .last_frame_time =current_time 

        frame_data ['detected_texts']=self .stable_detections 


        for result in indian_results :
            if result .get ('final_confidence',0 )>Config .CONFIDENCE_BOOST_THRESHOLD :
                self .plate_detected .emit (
                self .camera_id ,
                result ['text'],
                result 
                )


        if frame_data .get ('violations'):
            for violation in frame_data ['violations']:

                violation ['camera_id']=self .camera_id 
                violation ['timestamp_formatted']=datetime .now ().strftime ("%H:%M:%S")

                self .violation_alert .emit (violation )

        return frame_data 

    def _update_fps (self ):
        """Update FPS counter"""
        self .fps_counter +=1 
        if time .time ()-self .fps_start >=1.0 :
            self .current_fps =self .fps_counter 
            self .fps_counter =0 
            self .fps_start =time .time ()

    def stop (self ):
        """Stop camera processing"""
        self .running =False 
        if self .cap :
            self .cap .release ()
        self .ocr_processor .stop ()

    def add_emergency_vehicle (self ):
        """Add emergency vehicle from input"""
        plate_number =self .vehicle_input .text ().strip ().upper ()
        if not plate_number :
            return 


        validation_score =self ._validate_indian_plate_format (plate_number )
        if validation_score <0.8 :
            self .vehicle_input .setStyleSheet ("border: 2px solid red;")
            return 
        else :
            self .vehicle_input .setStyleSheet ("")


        if self .cameras and hasattr (self .cameras [0 ],'ocr_processor'):
            success =self .cameras [0 ].ocr_processor .add_emergency_vehicle (plate_number )
            if success :
                self .vehicle_input .clear ()
                self .load_vehicle_list ()

    def _validate_indian_plate_format (self ,text ):
        """Validate Indian plate format for UI"""
        if not text or len (text )<6 :
            return 0.0 


        indian_patterns =[
        r'^[A-Z]{2}[0-9]{2}[A-Z]{1,2}[0-9]{4}$',
        r'^[A-Z]{2}[0-9]{2}[A-Z]{1}[0-9]{4}$',
        r'^(AMB|POL|FIRE|ARMY|NAVY|AF)[0-9A-Z]{4,8}$',
        ]

        for pattern in indian_patterns :
            if re .match (pattern ,text ):
                return 1.0 

        return 0.0

    def add_recent_car (self ,camera_id ,vehicle_type ,license_plate ,confidence ,is_emergency ):
        """Add a car to recent cars list"""
        try :
            timestamp =datetime .now () .strftime ("%H:%M:%S")
            status = "EMERGENCY" if is_emergency else "NORMAL"
            
            car_data ={
                'timestamp':timestamp ,
                'camera_id':camera_id +1 ,
                'vehicle_type':vehicle_type ,
                'license_plate':license_plate ,
                'confidence':f"{confidence :.2f}",
                'status':status
            }
            
            self .recent_cars_data .insert (0 ,car_data )
            
            if len (self .recent_cars_data )>50 :
                self .recent_cars_data =self .recent_cars_data [:50 ]
            
            self .update_recent_cars_table ()
            
        except Exception as e :
            print (f"Error adding recent car: {e }")

    def update_recent_cars_table (self ):
        """Update the recent cars table"""
        try :
            self .recent_cars_table .setRowCount (len (self .recent_cars_data ))
            
            for row ,car in enumerate (self .recent_cars_data ):
                self .recent_cars_table .setItem (row ,0 ,QTableWidgetItem (car ['timestamp']))
                self .recent_cars_table .setItem (row ,1 ,QTableWidgetItem (str (car ['camera_id'])))
                self .recent_cars_table .setItem (row ,2 ,QTableWidgetItem (car ['vehicle_type']))
                self .recent_cars_table .setItem (row ,3 ,QTableWidgetItem (car ['license_plate']))
                self .recent_cars_table .setItem (row ,4 ,QTableWidgetItem (car ['confidence']))
                
                status_item =QTableWidgetItem (car ['status'])
                if car ['status']=='EMERGENCY':
                    status_item .setBackground (QColor (255 ,0 ,0 ))
                else :
                    status_item .setBackground (QColor (0 ,255 ,0 ))
                self .recent_cars_table .setItem (row ,5 ,status_item )
                
        except Exception as e :
            print (f"Error updating recent cars table: {e }")

    def export_violation_report (self ):
        """Export detailed violation report"""
        try :
            from datetime import datetime 
            import json 


            all_violations =[]
            for camera in self .cameras :
                if hasattr (camera ,'road_divider_manager'):
                    for track in camera .road_divider_manager .vehicle_tracks .values ():
                        all_violations .extend (track ['violations'])


            report ={
            'generated_at':datetime .now ().isoformat (),
            'total_violations':len (all_violations ),
            'violations':all_violations ,
            'statistics':self .get_violation_summary ()
            }


            filename =f"violation_report_{datetime .now ().strftime ('%Y%m%d_%H%M%S')}.json"
            with open (filename ,'w')as f :
                json .dump (report ,f ,indent =2 )

            print (f"Violation report exported: {filename }")

        except Exception as e :
            print (f"Error exporting violation report: {e }")

    def update_violation_stats (self ):
        """Update violation statistics display"""
        try :

            total_violations =0 
            center_crossings =0 
            rapid_changes =0 
            erratic_movement =0 

            for camera in self .cameras :
                if hasattr (camera ,'road_divider_manager'):
                    stats =camera .road_divider_manager .get_violation_stats ()
                    total_violations +=stats ['total_violations']

                    violation_types =stats .get ('violation_types',{})
                    center_crossings +=violation_types .get ('CENTER_DIVIDER_CROSSING',0 )
                    rapid_changes +=violation_types .get ('RAPID_LANE_CHANGES',0 )
                    erratic_movement +=violation_types .get ('ERRATIC_MOVEMENT',0 )


            if hasattr (self ,'total_violations_label'):
                self .total_violations_label .setText (f"Total Violations: {total_violations }")
                self .center_crossings_label .setText (f"Center Divider Crossings: {center_crossings }")
                self .rapid_changes_label .setText (f"Rapid Lane Changes: {rapid_changes }")
                self .erratic_movement_label .setText (f"Erratic Movement: {erratic_movement }")

        except Exception as e :
            print (f"Error updating violation stats: {e }")

    def get_violation_summary (self ):
        """Get summary of violation statistics"""
        summary ={
        'total_violations':0 ,
        'violation_types':{},
        'active_tracks':0 
        }

        for camera in self .cameras :
            if hasattr (camera ,'road_divider_manager'):
                stats =camera .road_divider_manager .get_violation_stats ()
                summary ['total_violations']+=stats ['total_violations']
                summary ['active_tracks']+=stats ['active_tracks']

                for vtype ,count in stats .get ('violation_types',{}).items ():
                    summary ['violation_types'][vtype ]=summary ['violation_types'].get (vtype ,0 )+count 

        return summary

    def switch_mode (self ,mode ):
        """Switch operation mode with visual feedback"""
        self .mode =mode 


        for camera in self .cameras :
            camera .set_mode (mode )

        print (f"🔄 Switched to {mode } mode")

    def start_camera_system (self ):
        """Initialize all cameras with enhanced processing"""
        for i in range (2 ):
            camera =HighPerformanceCamera (i ,self .mode )
            camera .frame_ready .connect (self .on_frame_ready )
            camera .plate_detected .connect (self .on_plate_detected )
            camera .violation_alert .connect (self .show_violation_alert )
            camera .start ()
            self .cameras .append (camera )

        print (" All camera systems started with ultra-accurate detection")

    def on_frame_ready (self ,camera_id ,frame ,data ):
        try :

            if hasattr (self ,'cameras')and len (self .cameras )>0 :
                camera =self .cameras [0 ]if self .cameras else None 
                if camera and hasattr (camera ,'road_divider_manager'):
                    frame =camera .road_divider_manager .draw_road_system (frame )


            if data .get ('violations'):
                for violation in data ['violations']:
                    if 'position'in violation :
                        x1 ,y1 ,x2 ,y2 =violation ['position']
                        cv2 .rectangle (frame ,(x1 ,y1 ),(x2 ,y2 ),(0 ,0 ,255 ),3 )
                        cv2 .putText (frame ,"VIOLATION",(x1 ,y1 -10 ),
                        cv2 .FONT_HERSHEY_SIMPLEX ,0.7 ,(0 ,0 ,255 ),2 )


            height ,width ,channel =frame .shape 
            bytes_per_line =3 *width 
            q_image =QImage (frame .data ,width ,height ,bytes_per_line ,QImage .Format_RGB888 ).rgbSwapped ()
            pixmap =QPixmap .fromImage (q_image )


            scaled =pixmap .scaled (400 ,300 ,Qt .KeepAspectRatio )
            if camera_id <len (self .camera_displays ):
                self .camera_displays [camera_id ].setPixmap (scaled )


            plates =len (data .get ('detected_texts',[]))
            violations =len (data .get ('violations',[]))
            fps =data .get ('fps',0 )
            self .stats_label .setText (f"Plates: {plates } | Violations: {violations } | FPS: {fps }")

        except Exception as e :
            print (f"Display error: {e }")

    def draw_sophisticated_overlays (self ,frame ,data ):
        """Draw sophisticated detection overlays"""


        if hasattr (self ,'cameras')and len (self .cameras )>0 :

            camera =self .cameras [0 ]if self .cameras else None 
            if camera and hasattr (camera ,'road_divider_manager'):
                frame =camera .road_divider_manager .draw_road_system (frame )

        if data .get ('vehicle_detections'):
            for det in data ['vehicle_detections']:
                x1 ,y1 ,x2 ,y2 ,conf ,cls =det 
                
                vehicle_colors ={
                2 :(0 ,255 ,0 ),   # car
                3 :(255 ,0 ,255 ), # motorcycle  
                5 :(0 ,255 ,255 ), # bus
                7 :(255 ,255 ,0 )  # truck
                }
                
                color =vehicle_colors .get (int (cls ),(128 ,128 ,128 ))
                
                cv2 .rectangle (frame ,(x1 ,y1 ),(x2 ,y2 ),color ,2 )
                cv2 .rectangle (frame ,(x1 ,y1 -25 ),(x2 ,y1 ),color ,-1 )
                
                vehicle_type =['','','car','motorcycle','','bus','','truck'][int (cls )]if int (cls )<8 else 'vehicle'
                label =f"{vehicle_type .upper ()} {conf :.2f}"
                cv2 .putText (frame ,label ,(x1 +2 ,y1 -8 ),
                cv2 .FONT_HERSHEY_SIMPLEX ,0.5 ,(0 ,0 ,0 ),1 )

        if data .get ('violations'):
            for violation in data ['violations']:
                if violation .get ('type')=='CENTER_DIVIDER_CROSSING'and 'position'in violation :
                    x1 ,y1 ,x2 ,y2 =violation ['position']

                    color =(0 ,0 ,128 )
                    cv2 .rectangle (frame ,(x1 -8 ,y1 -8 ),(x2 +8 ,y2 +8 ),color ,6 )

                    cv2 .rectangle (frame ,(x1 -8 ,max (0 ,y1 -40 )),(x2 +8 ,y1 -8 ),color ,-1 )
                    cv2 .putText (frame ,"RASH LANE CHANGING",(x1 ,max (15 ,y1 -16 )),
                    cv2 .FONT_HERSHEY_SIMPLEX ,0.7 ,(255 ,255 ,255 ),2 )


        for plate in data .get ('plates',[]):
            x1 ,y1 ,x2 ,y2 =plate ['bbox']
            confidence =plate ['confidence']
            method =plate .get ('method','unknown')


            method_colors ={
            'yolo_primary':(0 ,255 ,255 ),
            'yolo_backup':(255 ,255 ,0 ),
            'opencv_edge':(255 ,165 ,0 ),
            'opencv_morph':(255 ,0 ,255 ),
            'contour':(0 ,255 ,0 )
            }

            color =method_colors .get (method ,(128 ,128 ,128 ))


            cv2 .rectangle (frame ,(x1 ,y1 ),(x2 ,y2 ),color ,2 )
            cv2 .rectangle (frame ,(x1 ,y1 -25 ),(x2 ,y1 ),color ,-1 )


            label =f"{method .upper ()} {confidence :.2f}"
            cv2 .putText (frame ,label ,(x1 +2 ,y1 -8 ),
            cv2 .FONT_HERSHEY_SIMPLEX ,0.5 ,(0 ,0 ,0 ),1 )


        for text_result in data .get ('detected_texts',[]):
            x1 ,y1 ,x2 ,y2 =text_result ['bbox']
            text =text_result ['text']
            is_emergency =text_result ['is_emergency']
            validation_score =text_result .get ('validation_score',0.0 )


            if is_emergency :
                color =(0 ,0 ,255 )
                thickness =4 
            elif validation_score >0.8 :
                color =(0 ,255 ,0 )
                thickness =3 
            else :
                color =(255 ,255 ,255 )
                thickness =2 


            cv2 .rectangle (frame ,(x1 ,y1 ),(x2 ,y2 ),color ,thickness )
            cv2 .putText (frame ,text ,(x1 ,y1 -35 ),
            cv2 .FONT_HERSHEY_SIMPLEX ,0.8 ,color ,2 )


            score_text =f"Q:{validation_score :.1f}"
            cv2 .putText (frame ,score_text ,(x1 ,y2 +20 ),
            cv2 .FONT_HERSHEY_SIMPLEX ,0.5 ,color ,1 )

            if is_emergency :
                cv2 .putText (frame ,"EMERGENCY",(x1 ,y2 +40 ),
                cv2 .FONT_HERSHEY_SIMPLEX ,0.7 ,(0 ,0 ,255 ),2 )


        mode_color =(255 ,165 ,0 )if data .get ('mode')=='TEST'else (0 ,128 ,255 )
        cv2 .putText (frame ,f"MODE: {data .get ('mode','UNKNOWN')}",
        (10 ,30 ),cv2 .FONT_HERSHEY_SIMPLEX ,0.8 ,mode_color ,2 )

        accuracy =data .get ('accuracy_score',0.0 )
        acc_color =(0 ,255 ,0 )if accuracy >0.7 else (255 ,255 ,0 )if accuracy >0.4 else (255 ,0 ,0 )
        cv2 .putText (frame ,f"ACCURACY: {accuracy :.1%}",
        (10 ,60 ),cv2 .FONT_HERSHEY_SIMPLEX ,0.7 ,acc_color ,2 )

        return frame 

    def update_camera_stats (self ,camera_id ,data ):
        """Update ADAPTIVE camera statistics"""
        if camera_id <len (self .stats_displays ):
            stats =self .stats_displays [camera_id ]


            fps =data .get ('fps',0 )
            stats ['fps'].setText (f"{fps :.1f}")


            plates_count =len (data .get ('plates',[]))
            stats ['plates'].setText (str (plates_count ))


            performance_mode =data .get ('performance_mode','NORMAL')
            stats ['performance'].setText (performance_mode )


            accuracy_score =data .get ('accuracy_score',0.0 )
            stats ['accuracy'].setText (f"{accuracy_score :.1%}")
            self .accuracy_scores [camera_id ]=accuracy_score 


            emergency_count =len ([t for t in data .get ('detected_texts',[])if t .get ('is_emergency',False )])
            stats ['emergency'].setText (str (emergency_count ))


            if 'violations'in stats :
                violation_count =len (data .get ('violations',[]))
                stats ['violations'].setText (str (violation_count ))


                if 'violation_stats'in stats and data .get ('violation_stats'):
                    vstats =data ['violation_stats']
                    total_violations =vstats .get ('total_violations',0 )
                    stats ['violation_stats'].setText (f"Total: {total_violations }")

    def on_plate_detected (self ,camera_id ,plate_text ,plate_data ):
        """Handle sophisticated plate detection"""
        self .total_plates_detected +=1 


        row =self .detection_table .rowCount ()
        self .detection_table .insertRow (row )

        timestamp =datetime .now ().strftime ("%H:%M:%S.%f")[:-3 ]
        camera_name =f"CAM-{camera_id +1 }"
        confidence =f"{plate_data .get ('combined_confidence',0.0 ):.1%}"
        method =plate_data .get ('method','unknown').upper ()
        plate_type =" EMERGENCY"if plate_data .get ('is_emergency',False )else "CIVILIAN"


        items =[
        QTableWidgetItem (timestamp ),
        QTableWidgetItem (camera_name ),
        QTableWidgetItem (plate_text ),
        QTableWidgetItem (confidence ),
        QTableWidgetItem (method ),
        QTableWidgetItem (plate_type )
        ]

        for col ,item in enumerate (items ):
            self .detection_table .setItem (row ,col ,item )

        if plate_data .get ('is_emergency',False ):
            self .emergency_count +=1 

            self .add_recent_car (camera_id ,"CAR" ,plate_text ,plate_data .get ('combined_confidence',0.0 ),plate_data .get ('is_emergency',False ))
            for col in range (6 ):
                item =self .detection_table .item (row ,col )
                if item :
                    item .setBackground (QColor ("#FFE6E6"))

            self .update_emergency_status (camera_id ,plate_text )
        else:
            self .add_recent_car (camera_id ,"CAR" ,plate_text ,plate_data .get ('combined_confidence',0.0 ),plate_data .get ('is_emergency',False ))

        self .detection_table .scrollToBottom ()
        if row >200 :
            self .detection_table .removeRow (0 )

        print (f"🎯 DETECTED: {plate_text } on Camera {camera_id +1 } (Confidence: {confidence })")

    def update_adaptive_ui (self ):
        """Update ADAPTIVE UI with smart performance monitoring"""

        self .total_plates_metric .setText (str (self .total_plates_detected ))
        self .emergency_metric .setText (str (self .emergency_count ))


        if self .accuracy_scores :
            avg_accuracy =sum (self .accuracy_scores )/len (self .accuracy_scores )
            self .accuracy_metric .setText (f"{avg_accuracy :.1%}")


        if hasattr (self ,'violations_metric'):
            total_violations =0 
            for camera in self .cameras :
                if hasattr (camera ,'road_divider_manager'):
                    stats =camera .road_divider_manager .get_violation_stats ()
                    total_violations +=stats ['total_violations']
            self .violations_metric .setText (str (total_violations ))


        current_time =datetime .now ().strftime ("%Y-%m-%d %H:%M:%S")
        if hasattr (self ,'system_time'):
            self .system_time .setText (current_time )


        if hasattr (self ,'detection_rate'):
            elapsed =max (1 ,time .time ()-getattr (self ,'start_time',time .time ()))
            rate =self .total_plates_detected *60 /elapsed 
            self .detection_rate .setText (f"{rate :.2f} plates/min")


        if hasattr (self ,'system_status'):
            avg_accuracy =sum (self .accuracy_scores )/len (self .accuracy_scores )if self .accuracy_scores else 0 
            if avg_accuracy >0.8 :
                self .system_status .setText (" ADAPTIVE OPTIMAL")
            elif avg_accuracy >0.6 :
                self .system_status .setText (" HIGH PERFORMANCE")
            elif avg_accuracy >0.4 :
                self .system_status .setText (" NORMAL MODE")
            else :
                self .system_status .setText (" SPEED MODE")

            print (f"Error updating timer: {e }")

        except Exception as e :

    def detect_plates_ultra_accurate (self ,frame ):
        """Ultra-accurate multi-algorithm plate detection"""
        all_detections =[]


        if self .primary_model :
            yolo_detections =self ._yolo_detection (frame ,self .primary_model ,"primary")
            all_detections .extend (yolo_detections )


        if self .backup_model and len (all_detections )<3 :
            backup_detections =self ._yolo_detection (frame ,self .backup_model ,"backup")
            all_detections .extend (backup_detections )


        cv_detections =self ._enhanced_opencv_detection (frame )
        all_detections .extend (cv_detections )


        merged_detections =self ._merge_detections (all_detections )
        validated_detections =self ._validate_detections (frame ,merged_detections )

        return validated_detections [:Config .MAX_PLATES_PER_FRAME ]

    def _yolo_detection (self ,frame ,model ,model_type ):
        """YOLO-based detection with enhanced post-processing"""
        try :
            results =model (frame ,
            imgsz =Config .AI_INPUT_SIZE ,
            conf =Config .CONFIDENCE_THRESHOLD ,
            iou =Config .NMS_THRESHOLD ,
            verbose =False )

            detections =[]
            if len (results )>0 and hasattr (results [0 ],'boxes')and results [0 ].boxes is not None :
                boxes =results [0 ].boxes 

                for box in boxes :
                    x1 ,y1 ,x2 ,y2 =box .xyxy [0 ].cpu ().numpy ()
                    confidence =box .conf [0 ].cpu ().numpy ()


                    if self ._validate_bbox_geometry (x1 ,y1 ,x2 ,y2 ,frame .shape ):
                        roi =frame [int (y1 ):int (y2 ),int (x1 ):int (x2 )]
                        enhanced_roi =self ._enhance_roi_for_ocr (roi )

                        detections .append ({
                        'bbox':(int (x1 ),int (y1 ),int (x2 ),int (y2 )),
                        'confidence':float (confidence ),
                        'roi':enhanced_roi ,
                        'method':f'yolo_{model_type }',
                        'quality_score':self ._calculate_roi_quality (enhanced_roi )
                        })

            return detections 

        except Exception as e :
            return []

    def _enhanced_opencv_detection (self ,frame ):
        """Enhanced OpenCV detection with multiple filters"""
        detections =[]


        gray =cv2 .cvtColor (frame ,cv2 .COLOR_BGR2GRAY )


        edges =cv2 .Canny (gray ,30 ,200 )
        edges =cv2 .dilate (edges ,self .morphology_kernel ,iterations =1 )
        edges =cv2 .erode (edges ,self .morphology_kernel ,iterations =1 )

        contours ,_ =cv2 .findContours (edges ,cv2 .RETR_EXTERNAL ,cv2 .CHAIN_APPROX_SIMPLE )

        for contour in contours :
            x ,y ,w ,h =cv2 .boundingRect (contour )

            if self ._is_plate_like_region (w ,h ,cv2 .contourArea (contour )):
                roi =frame [y :y +h ,x :x +w ]
                enhanced_roi =self ._enhance_roi_for_ocr (roi )
                quality =self ._calculate_roi_quality (enhanced_roi )

                if quality >0.3 :
                    detections .append ({
                    'bbox':(x ,y ,x +w ,y +h ),
                    'confidence':min (0.8 ,quality ),
                    'roi':enhanced_roi ,
                    'method':'opencv_edge',
                    'quality_score':quality 
                    })

        return detections 

    def _enhance_roi_for_ocr (self ,roi ):
        """Advanced ROI enhancement for better OCR accuracy"""
        if roi is None or roi .size ==0 :
            return None 

        try :

            if len (roi .shape )==3 :
                gray =cv2 .cvtColor (roi ,cv2 .COLOR_BGR2GRAY )
            else :
                gray =roi .copy ()


            height ,width =gray .shape 
            if width <300 :
                scale =300 /width 
                new_width =int (width *scale )
                new_height =int (height *scale )
                gray =cv2 .resize (gray ,(new_width ,new_height ),interpolation =cv2 .INTER_CUBIC )




            enhanced =cv2 .GaussianBlur (gray ,(3 ,3 ),0 )


            clahe =cv2 .createCLAHE (clipLimit =3.0 ,tileGridSize =(8 ,8 ))
            enhanced =clahe .apply (enhanced )


            enhanced =cv2 .bilateralFilter (enhanced ,9 ,75 ,75 )


            kernel =cv2 .getStructuringElement (cv2 .MORPH_RECT ,(2 ,2 ))
            enhanced =cv2 .morphologyEx (enhanced ,cv2 .MORPH_CLOSE ,kernel )


            _ ,enhanced =cv2 .threshold (enhanced ,0 ,255 ,cv2 .THRESH_BINARY +cv2 .THRESH_OTSU )

            return enhanced 

        except Exception as e :
            return roi 

    def _calculate_roi_quality (self ,roi ):
        """Calculate ROI quality score for filtering"""
        if roi is None or roi .size ==0 :
            return 0.0 

        try :

            if len (roi .shape )==3 :
                gray =cv2 .cvtColor (roi ,cv2 .COLOR_BGR2GRAY )
            else :
                gray =roi 


            height ,width =gray .shape 


            size_score =min (1.0 ,(width *height )/10000 )


            contrast =gray .std ()
            contrast_score =min (1.0 ,contrast /50 )


            edges =cv2 .Canny (gray ,50 ,150 )
            edge_density =np .sum (edges >0 )/(width *height )
            edge_score =min (1.0 ,edge_density *10 )


            aspect_ratio =width /height if height >0 else 0 
            if 2.0 <=aspect_ratio <=5.0 :
                aspect_score =1.0 
            else :
                aspect_score =0.5 


            quality =(size_score *0.3 +contrast_score *0.3 +
            edge_score *0.3 +aspect_score *0.1 )

            return quality 

        except :
            return 0.0 

    def _validate_bbox_geometry (self ,x1 ,y1 ,x2 ,y2 ,frame_shape ):
        """Validate bounding box geometry"""
        height ,width =frame_shape [:2 ]


        if x1 <0 or y1 <0 or x2 >=width or y2 >=height :
            return False 


        w ,h =x2 -x1 ,y2 -y1 
        if w <40 or h <15 :
            return False 


        if w >width *0.8 or h >height *0.8 :
            return False 


        aspect_ratio =w /h if h >0 else 0 
        if not (1.5 <=aspect_ratio <=8.0 ):
            return False 

        return True 

    def _is_plate_like_region (self ,width ,height ,area ):
        """Check if region has plate-like characteristics"""
        if height ==0 :
            return False 

        aspect_ratio =width /height 


        if width <60 or height <20 :
            return False 


        if not (2.0 <=aspect_ratio <=6.0 ):
            return False 


        bbox_area =width *height 
        if bbox_area >0 and area /bbox_area <0.3 :
            return False 

        return True 

    def _merge_detections (self ,all_detections ):
        """Merge overlapping detections from different methods"""
        if not all_detections :
            return []


        all_detections .sort (key =lambda x :x ['confidence'],reverse =True )

        merged =[]

        for detection in all_detections :
            is_duplicate =False 

            for existing in merged :
                if self ._calculate_iou (detection ['bbox'],existing ['bbox'])>0.5 :

                    if detection ['confidence']>existing ['confidence']:
                        merged .remove (existing )
                        merged .append (detection )
                    is_duplicate =True 
                    break 

            if not is_duplicate :
                merged .append (detection )

        return merged 

    def _calculate_iou (self ,bbox1 ,bbox2 ):
        """Calculate Intersection over Union"""
        x1_1 ,y1_1 ,x2_1 ,y2_1 =bbox1 
        x1_2 ,y1_2 ,x2_2 ,y2_2 =bbox2 


        x1_i =max (x1_1 ,x1_2 )
        y1_i =max (y1_1 ,y1_2 )
        x2_i =min (x2_1 ,x2_2 )
        y2_i =min (y2_1 ,y2_2 )

        if x2_i <=x1_i or y2_i <=y1_i :
            return 0.0 

        intersection =(x2_i -x1_i )*(y2_i -y1_i )


        area1 =(x2_1 -x1_1 )*(y2_1 -y1_1 )
        area2 =(x2_2 -x1_2 )*(y2_2 -y1_2 )
        union =area1 +area2 -intersection 

        return intersection /union if union >0 else 0.0 

    def _validate_detections (self ,frame ,detections ):
        """Final validation of detections"""
        validated =[]

        for detection in detections :

            if detection ['quality_score']<0.25 :
                continue 


            x1 ,y1 ,x2 ,y2 =detection ['bbox']
            w ,h =x2 -x1 ,y2 -y1 

            if w <50 or h <18 :
                continue 


            validated .append (detection )

        return validated

        """Start ADAPTIVE OCR worker threads"""
        self .running =True 


        for i in range (self .current_thread_count ):
            thread =threading .Thread (target =self ._adaptive_ocr_worker ,daemon =True )
            thread .start ()
            self .processing_threads .append (thread )

        print (f"🚀 Started {self .current_thread_count } adaptive OCR workers")

    def process_plate_async (self ,plate_data ,frame_id ,camera_id ):
        """Add plate for enhanced async OCR processing"""
        if not self .ocr_ready :
            return 

        try :
            self .ocr_queue .put_nowait ({
            'roi':plate_data ['roi'],
            'bbox':plate_data ['bbox'],
            'confidence':plate_data ['confidence'],
            'method':plate_data ['method'],
            'quality_score':plate_data ['quality_score'],
            'frame_id':frame_id ,
            'camera_id':camera_id ,
            'timestamp':time .time ()
            })
        except queue .Full :
            pass 

    def get_ocr_results (self ):
        """Get all available enhanced OCR results"""
        results =[]

        try :
            while True :
                result =self .result_queue .get_nowait ()
                results .append (result )
        except queue .Empty :
            pass 

        return results 

    def _adaptive_ocr_worker (self ):
        """ADAPTIVE OCR worker with smart processing"""
        while self .running :
            try :
                plate_data =self .ocr_queue .get (timeout =0.02 )


                settings =self .performance_monitor .get_adaptive_settings ()


                engine_to_use =self .ocr_engines [0 ]if self .ocr_engines else None 
                if not engine_to_use :
                    continue 

                engine_name ,engine =engine_to_use 

                try :

                    text_result =self ._extract_text_adaptive (
                    plate_data ['roi'],
                    engine ,
                    engine_name ,
                    settings 
                    )

                    if text_result :

                        final_result ={
                        'text':text_result ['text'],
                        'bbox':plate_data ['bbox'],
                        'confidence':plate_data ['confidence'],
                        'ocr_confidence':text_result ['confidence'],
                        'final_confidence':text_result ['confidence'],
                        'method':f"adaptive_{engine_name }",
                        'quality_score':plate_data ['quality_score'],
                        'frame_id':plate_data ['frame_id'],
                        'camera_id':plate_data ['camera_id'],
                        'timestamp':plate_data ['timestamp'],
                        'is_emergency':self ._is_emergency_plate (text_result ['text']),
                        'validation_score':self ._fast_validation (text_result ['text'])
                        }

                        try :
                            self .result_queue .put_nowait (final_result )
                        except queue .Full :
                            pass 

                except Exception as e :
                    continue 

            except queue .Empty :
                continue 
            except Exception as e :
                continue 

    def _extract_text_adaptive (self ,roi ,engine ,engine_name ,settings ):
        """ADAPTIVE text extraction with smart optimization"""
        if roi is None or roi .size ==0 :
            return None 

        try :

            if not settings .get ('skip_preprocessing',False ):
                roi =self ._smart_preprocess (roi ,settings )


            results =engine .readtext (
            roi ,
            detail =1 ,
            paragraph =False ,
            width_ths =0.7 if settings .get ('fast_mode',True )else 0.5 ,
            height_ths =0.7 if settings .get ('fast_mode',True )else 0.5 
            )


            best_result =None 
            best_confidence =0 

            for bbox ,text ,confidence in results :
                if confidence >Config .OCR_MIN_CONFIDENCE :
                    clean_text =self ._fast_text_cleaning (text )
                    if clean_text and confidence >best_confidence :
                        best_result ={

        except Exception as e :

    def _fast_text_cleaning (self ,text ):
        """FAST text cleaning optimized for Indian plates"""
        if not text :
            return None 


        text =text .upper ().strip ()
        text =re .sub (r'[^A-Z0-9]','',text )


        char_fixes ={
        'O':'0','0':'O',
        'I':'1','1':'I',
        'S':'5','5':'S',
        'B':'8','8':'B',
        'G':'6','6':'G',
        'Z':'2','2':'Z'
        }


        fixed_text =""
        for i ,char in enumerate (text ):
            if i <2 :
                if char .isdigit ()and char in char_fixes :
                    fixed_text +=char_fixes [char ]if char_fixes [char ].isalpha ()else char 
                else :
                    fixed_text +=char 
            elif 2 <=i <4 :
                if char .isalpha ()and char in char_fixes :
                    fixed_text +=char_fixes [char ]if char_fixes [char ].isdigit ()else char 
                else :
                    fixed_text +=char 
            elif 4 <=i <6 :
                if char .isdigit ()and char in char_fixes :
                    fixed_text +=char_fixes [char ]if char_fixes [char ].isalpha ()else char 
                else :
                    fixed_text +=char 
            else :
                if char .isalpha ()and char in char_fixes :
                    fixed_text +=char_fixes [char ]if char_fixes [char ].isdigit ()else char 
                else :
                    fixed_text +=char 

        return fixed_text if len (fixed_text )>=6 else None 

    def _fast_validation (self ,text ):
        """FAST validation for Indian license plates"""
        if not text or len (text )<6 :
            return 0.0 

        score =0.0 


        indian_patterns =[

        r'^[A-Z]{2}[0-9]{2}[A-Z]{1,2}[0-9]{4}$',

        r'^[A-Z]{2}[0-9]{2}[A-Z]{1}[0-9]{4}$',

        r'^[A-Z]{2}[0-9]{2}[A-Z]{2}[0-9]{3}[A-Z]$',

        r'^(AMB|POL|FIRE|ARMY|NAVY|AF)[0-9A-Z]{4,8}$',

        r'^[A-Z]{2}[0-9]{1,2}[A-Z]{1,3}[0-9]{1,4}$'
        ]


        for pattern in indian_patterns :
            if re .match (pattern ,text ):
                score +=0.8 
                break 
        else :

            if re .match (r'^[A-Z]{2}',text ):
                score +=0.3 
            if re .search (r'[0-9]{4}$',text ):
                score +=0.3 
            if 8 <=len (text )<=10 :
                score +=0.2 


        indian_state_codes ={
        'DL','MH','KA','TN','AP','TS','UP','MP','RJ','GJ',
        'WB','OR','BR','JH','HR','PB','HP','UK','CH','GA',
        'SK','ML','MN','TR','NL','MZ','AR','AS','AN','JK',
        'LA','LD','PY','DN','DD','CG'
        }

        if len (text )>=2 and text [:2 ]in indian_state_codes :
            score +=0.2 

        return min (1.0 ,score )

    def _validate_plate_text (self ,text ):
        """Enhanced plate text validation with scoring"""
        if not text or len (text )<6 :
            return 0.0 

        score =0.0 


        if 8 <=len (text )<=12 :
            score +=0.3 
        elif 6 <=len (text )<8 :
            score +=0.2 
        else :
            score +=0.1 


        patterns =[
        r'^[A-Z]{2}[0-9]{2}[A-Z]{1,2}[0-9]{4}$',
        r'^[A-Z]{2}[0-9]{1,2}[A-Z]{1,3}[0-9]{1,4}$',
        r'^[A-Z]{3}[0-9]{3,4}$',
        r'^[0-9]{2}[A-Z]{2}[0-9]{4}$',
        ]

        for pattern in patterns :
            if re .match (pattern ,text ):
                score +=0.4 
                break 
        else :

            if re .match (r'^[A-Z]{2}',text ):
                score +=0.2 
            if re .search (r'[0-9]{3,4}$',text ):
                score +=0.2 


        letters =sum (1 for c in text if c .isalpha ())
        numbers =sum (1 for c in text if c .isdigit ())

        if 2 <=letters <=6 and numbers >=3 :
            score +=0.3 

        return min (1.0 ,score )

class RoadDividerManager :
    """Professional Road Divider and Lane Management System"""

    def __init__ (self ,frame_width ,frame_height ):
        self .frame_width =frame_width 
        self .frame_height =frame_height 
        self .lanes =[]
        self .divider_lines =[]
        self .lane_boundaries =[]
        self .setup_road_layout ()


        self .vehicle_tracks ={}
        self .violation_history ={}

    def setup_road_layout (self ):
        """Setup professional road layout with lanes and dividers"""
        if not Config .ENABLE_ROAD_DIVIDERS :
            return 


        usable_width =int (self .frame_width *0.8 )
        start_x =(self .frame_width -usable_width )//2 
        lane_width =usable_width //Config .NUM_LANES 


        for i in range (Config .NUM_LANES +1 ):
            x_pos =start_x +(i *lane_width )
            self .lane_boundaries .append (x_pos )


        center_divider_x =start_x +(usable_width //2 )
        self .divider_lines .append ({
        'x':center_divider_x ,
        'type':'center_divider',
        'color':Config .DIVIDER_COLOR ,
        'thickness':Config .DIVIDER_THICKNESS 
        })


        for i in range (1 ,Config .NUM_LANES ):
            if i !=Config .NUM_LANES //2 :
                x_pos =start_x +(i *lane_width )
                self .divider_lines .append ({
                'x':x_pos ,
                'type':'lane_marker',
                'color':Config .LANE_LINE_COLOR ,
                'thickness':Config .LANE_LINE_THICKNESS 
                })


        for i in range (Config .NUM_LANES ):
            left_boundary =start_x +(i *lane_width )
            right_boundary =start_x +((i +1 )*lane_width )
            direction ='left_to_right'if i <Config .NUM_LANES //2 else 'right_to_left'

            self .lanes .append ({
            'id':i ,
            'left_boundary':left_boundary ,
            'right_boundary':right_boundary ,
            'center_x':(left_boundary +right_boundary )//2 ,
            'direction':direction ,
            'violations':0 
            })

    def draw_road_system (self ,frame ):
        """Draw professional road dividers and lane markers"""
        if not Config .ENABLE_ROAD_DIVIDERS :
            return frame 


        overlay =frame .copy ()


        for divider in self .divider_lines :
            if divider ['type']=='center_divider':

                cv2 .line (overlay ,
                (divider ['x'],0 ),
                (divider ['x'],self .frame_height ),
                self ._hex_to_bgr (divider ['color']),
                divider ['thickness'])


                for y in range (0 ,self .frame_height ,40 ):
                    cv2 .rectangle (overlay ,
                    (divider ['x']-2 ,y ),
                    (divider ['x']+2 ,y +20 ),
                    self ._hex_to_bgr ("#FFFF00"),-1 )
            else :

                self ._draw_dashed_line (overlay ,divider ['x'],
                self ._hex_to_bgr (divider ['color']),
                divider ['thickness'])


        if hasattr (Config ,'SHOW_LANE_NUMBERS')and getattr (Config ,'SHOW_LANE_NUMBERS',False ):
            for lane in self .lanes :
                cv2 .putText (overlay ,f"L{lane ['id']}",
                (lane ['center_x']-10 ,30 ),
                cv2 .FONT_HERSHEY_SIMPLEX ,0.6 ,
                self ._hex_to_bgr ("#00FF00"),2 )

        frame =cv2 .addWeighted (frame ,0.85 ,overlay ,0.15 ,0 )
        return frame 

    def _draw_dashed_line (self ,frame ,x ,color ,thickness ):
        """Draw dashed lane marker lines"""
        dash_length =30 
        gap_length =20 
        y =0 

        while y <self .frame_height :
            cv2 .line (frame ,(x ,y ),(x ,min (y +dash_length ,self .frame_height )),
            color ,thickness )
            y +=dash_length +gap_length 

    def _hex_to_bgr (self ,hex_color ):
        """Convert hex color to BGR tuple"""
        hex_color =hex_color .lstrip ('#')
        return tuple (int (hex_color [i :i +2 ],16 )for i in (4 ,2 ,0 ))

    def detect_lane_violation (self ,vehicle_bbox ,vehicle_id ,timestamp ):
        """Detect if vehicle crosses lane dividers (rash driving)"""
        if not Config .RASH_DRIVING_ENABLED :
            return None 

        x1 ,y1 ,x2 ,y2 =vehicle_bbox 
        vehicle_center_x =(x1 +x2 )//2 
        vehicle_center_y =(y1 +y2 )//2 


        current_lane =self ._get_vehicle_lane (vehicle_center_x )
        if current_lane is None :
            return None 


        if vehicle_id not in self .vehicle_tracks :
            self .vehicle_tracks [vehicle_id ]={
            'positions':[],
            'lanes':[],
            'violations':[],
            'last_violation_time':0 
            }

        track =self .vehicle_tracks [vehicle_id ]
        track ['positions'].append ((vehicle_center_x ,vehicle_center_y ,timestamp ))
        track ['lanes'].append (current_lane )


        if len (track ['positions'])>30 :
            track ['positions']=track ['positions'][-30 :]
            track ['lanes']=track ['lanes'][-30 :]


        violation =self ._analyze_lane_behavior (vehicle_id ,vehicle_bbox ,timestamp )

        if violation :
            track ['violations'].append (violation )
            track ['last_violation_time']=timestamp 


            self ._log_lane_violation (vehicle_id ,violation ,timestamp )

        return violation 

    def _get_vehicle_lane (self ,vehicle_center_x ):
        """Determine which lane the vehicle is in"""
        for lane in self .lanes :
            if lane ['left_boundary']<=vehicle_center_x <=lane ['right_boundary']:
                return lane ['id']
        return None 

    def _analyze_lane_behavior (self ,vehicle_id ,vehicle_bbox ,timestamp ):
        """Analyze vehicle behavior for lane violations"""
        track =self .vehicle_tracks [vehicle_id ]

        if len (track ['positions'])<5 :
            return None 


        violation =self ._check_center_divider_crossing (vehicle_id ,vehicle_bbox ,timestamp )
        if violation :
            return violation 


        violation =self ._check_rapid_lane_changes (vehicle_id ,timestamp )
        if violation :
            return violation 


        violation =self ._check_erratic_movement (vehicle_id ,timestamp )
        if violation :
            return violation 

        return None 

    def _check_center_divider_crossing (self ,vehicle_id ,vehicle_bbox ,timestamp ):
        """Check if vehicle crosses the center divider"""
        x1 ,y1 ,x2 ,y2 =vehicle_bbox 
        track =self .vehicle_tracks [vehicle_id ]


        center_divider =None 
        for divider in self .divider_lines :
            if divider ['type']=='center_divider':
                center_divider =divider ['x']
                break 

        if center_divider is None :
            return None 


        buffer =Config .LANE_VIOLATION_BUFFER 
        if (x1 -buffer )<=center_divider <=(x2 +buffer ):

            if len (track ['lanes'])>=3 :
                recent_lanes =track ['lanes'][-3 :]
                left_side_lanes =[i for i in recent_lanes if i <Config .NUM_LANES //2 ]
                right_side_lanes =[i for i in recent_lanes if i >=Config .NUM_LANES //2 ]

                if left_side_lanes and right_side_lanes :
                    return {
                    'type':'CENTER_DIVIDER_CROSSING',
                    'severity':'HIGH',
                    'description':'Vehicle crossed center divider - DANGEROUS!',
                    'position':(x1 ,y1 ,x2 ,y2 ),
                    'timestamp':timestamp ,
                    'lane_from':min (recent_lanes ),
                    'lane_to':max (recent_lanes )
                    }
        return None 

    def _check_rapid_lane_changes (self ,vehicle_id ,timestamp ):
        """Detect rapid/unsafe lane changes"""
        track =self .vehicle_tracks [vehicle_id ]

        if len (track ['lanes'])<10 :
            return None 

        recent_lanes =track ['lanes'][-10 :]
        recent_times =[pos [2 ]for pos in track ['positions'][-10 :]]


        lane_changes =0 
        for i in range (1 ,len (recent_lanes )):
            if recent_lanes [i ]!=recent_lanes [i -1 ]:
                lane_changes +=1 


        time_span =recent_times [-1 ]-recent_times [0 ]
        if time_span >0 and lane_changes >=3 and time_span <5.0 :
            return {
            'type':'RAPID_LANE_CHANGES',
            'severity':'MEDIUM',
            'description':f'{lane_changes } lane changes in {time_span :.1f}s - Rash driving!',
            'timestamp':timestamp ,
            'lane_changes':lane_changes ,
            'time_span':time_span 
            }
        return None 

    def _check_erratic_movement (self ,vehicle_id ,timestamp ):
        """Detect erratic/zigzag movement patterns"""
        track =self .vehicle_tracks [vehicle_id ]

        if len (track ['positions'])<8 :
            return None 

        positions =track ['positions'][-8 :]


        direction_changes =0 
        prev_direction =None 

        for i in range (1 ,len (positions )):
            dx =positions [i ][0 ]-positions [i -1 ][0 ]
            if abs (dx )>5 :
                current_direction ='left'if dx <0 else 'right'
                if prev_direction and prev_direction !=current_direction :
                    direction_changes +=1 
                prev_direction =current_direction 


        if direction_changes >=4 :
            return {
            'type':'ERRATIC_MOVEMENT',
            'severity':'MEDIUM',
            'description':'Erratic zigzag movement detected - Possible rash driving!',
            'timestamp':timestamp ,
            'direction_changes':direction_changes 
            }
        return None 

    def _log_lane_violation (self ,vehicle_id ,violation ,timestamp ):
        """Log lane violation to file"""
        try :
            os .makedirs ("rash_logs",exist_ok =True )

            log_entry ={
            'timestamp':datetime .now ().isoformat (),
            'vehicle_id':vehicle_id ,
            'violation_type':violation ['type'],
            'severity':violation ['severity'],
            'description':violation ['description'],
            'details':violation 
            }

            log_file =f"rash_logs/lane_violations_{datetime .now ().strftime ('%Y%m%d')}.json"


            logs =[]
            if os .path .exists (log_file ):
                try :
                    with open (log_file ,'r')as f :
                        logs =json .load (f )
                except :
                    logs =[]

            logs .append (log_entry )


            with open (log_file ,'w')as f :
                json .dump (logs ,f ,indent =2 )

            print (f"🚨 LANE VIOLATION LOGGED: {violation ['type']} - {violation ['description']}")

        except Exception as e :
            print (f"Error logging lane violation: {e }")

    def get_violation_stats (self ):
        """Get lane violation statistics"""
        total_violations =0

                        conf =float (box .conf [0 ].cpu ().numpy ())if hasattr (box ,'conf')else 0.0 
                        dets .append ((x1 ,y1 ,x2 ,y2 ,conf ,cls ))
            return dets 
        except Exception :
            return []

class SmartTrafficDetector :
    """Enhanced AI-powered traffic violation and emergency detection"""
    def __init__ (self ):
        self .violation_memory ={}
        self .emergency_memory ={}
        self .saved_violations =set ()
        self .plate_history ={}

    def detect_emergency_vehicle (self ,frame ,detections ):
        """Detect emergency vehicles using AI and visual cues"""
        emergency_vehicles =[]

        for detection in detections :
            x1 ,y1 ,x2 ,y2 ,conf ,cls =detection 
            if conf >0.5 :

                roi =frame [int (y1 ):int (y2 ),int (x1 ):int (x2 )]


                if self ._is_emergency_vehicle (roi ,cls ):
                    emergency_vehicles .append ({
                    'bbox':(x1 ,y1 ,x2 ,y2 ),
                    'confidence':conf ,
                    'type':'emergency'
                    })

        return emergency_vehicles 

    def _is_emergency_vehicle (self ,roi ,vehicle_class ):
        """Analyze ROI to determine if it's an emergency vehicle"""
        if roi .size ==0 :
            return False 


        hsv =cv2 .cvtColor (roi ,cv2 .COLOR_BGR2HSV )


        red_mask1 =cv2 .inRange (hsv ,np .array ([0 ,50 ,50 ]),np .array ([10 ,255 ,255 ]))
        red_mask2 =cv2 .inRange (hsv ,np .array ([170 ,50 ,50 ]),np .array ([180 ,255 ,255 ]))
        blue_mask =cv2 .inRange (hsv ,np .array ([100 ,50 ,50 ]),np .array ([130 ,255 ,255 ]))

        red_pixels =cv2 .countNonZero (red_mask1 )+cv2 .countNonZero (red_mask2 )
        blue_pixels =cv2 .countNonZero (blue_mask )

        total_pixels =roi .shape [0 ]*roi .shape [1 ]
        emergency_ratio =(red_pixels +blue_pixels )/total_pixels 


        return emergency_ratio >0.05 or vehicle_class ==AMBULANCE_CLASS_ID 

    def detect_violations (self ,frame ,detections ,traffic_state ):
        """Detect traffic violations including red light running"""
        violations =[]

        for detection in detections :
            x1 ,y1 ,x2 ,y2 ,conf ,cls =detection 
            if conf >0.5 :
                center_x =(x1 +x2 )/2 
                center_y =(y1 +y2 )/2 


                if traffic_state =='red'and self ._is_in_intersection (center_x ,center_y ,frame .shape ):
                    violations .append ({
                    'type':'red_light_violation',
                    'bbox':(x1 ,y1 ,x2 ,y2 ),
                    'position':(center_x ,center_y ),
                    'confidence':conf 
                    })


                if self ._detect_rash_driving (center_x ,center_y ,frame .shape ):
                    violations .append ({
                    'type':'rash_driving',
                    'bbox':(x1 ,y1 ,x2 ,y2 ),
                    'position':(center_x ,center_y ),
                    'confidence':conf 
                    })

        return violations 

    def _is_in_intersection (self ,x ,y ,frame_shape ):
        """Check if vehicle is in intersection area"""
        h ,w =frame_shape [:2 ]
        intersection_zone ={
        'x_min':w *0.3 ,
        'x_max':w *0.7 ,
        'y_min':h *0.3 ,
        'y_max':h *0.7 
        }
        return (intersection_zone ['x_min']<x <intersection_zone ['x_max']and 
        intersection_zone ['y_min']<y <intersection_zone ['y_max'])

    def _detect_rash_driving (self ,x ,y ,frame_shape ):
        """Enhanced rash driving detection with lane analysis"""




        if hasattr (self ,'prev_positions'):
            if (x ,y )in self .prev_positions :

                movements =[abs (px -x )+abs (py -y )for px ,py in self .prev_positions [-5 :]]
                if len (movements )>3 :
                    avg_movement =sum (movements )/len (movements )
                    if avg_movement >Config .ERRATIC_MOVEMENT_THRESHOLD :
                        return True 
        else :
            self .prev_positions =[]

        self .prev_positions .append ((x ,y ))
        if len (self .prev_positions )>10 :
            self .prev_positions =self .prev_positions [-10 :]

        return False 

class EnhancedPlateProcessor :
    """Improved license plate detection and recognition"""
    def __init__ (self ):
        self .prev_plates ={}
        self .emergency_plates =set ()

    def process_plate (self ,plate_roi ):
        """Enhanced plate processing with better accuracy"""
        if plate_roi .size ==0 :
            return None 


        enhanced =self ._enhance_plate_image (plate_roi )
        text =self ._ocr_plate (enhanced )

        if text and self ._validate_plate (text ):
            return text 
        return None 

    def _enhance_plate_image (self ,plate_img ):
        """Advanced plate image enhancement"""
        gray =cv2 .cvtColor (plate_img ,cv2 .COLOR_BGR2GRAY )


        clahe =cv2 .createCLAHE (clipLimit =3.0 ,tileGridSize =(8 ,8 ))
        enhanced =clahe .apply (gray )


        denoised =cv2 .fastNlMeansDenoising (enhanced )


        _ ,thresh =cv2 .threshold (denoised ,0 ,255 ,cv2 .THRESH_BINARY +cv2 .THRESH_OTSU )

        return thresh 

    def _ocr_plate (self ,processed_plate ):
        """OCR processing with error correction"""
        if reader is None :
            return None 

        try :
            results =reader .readtext (processed_plate ,detail =0 ,paragraph =False )
            if results :
                text =''.join (results ).upper ()

                clean_text =re .sub (r'[^A-Z0-9]','',text )
                return clean_text 
        except :
            return None 
        return None 

    def _validate_plate (self ,text ):
        """Validate Indian license plate format"""
        if not text or len (text )<8 :
            return False 


        patterns =[
        r'^[A-Z]{2}[0-9]{2}[A-Z]{1,2}[0-9]{4}$',
        r'^[A-Z]{2}[0-9]{1,2}[A-Z]{1,3}[0-9]{1,4}$'
        ]

        return any (re .match (pattern ,text )for pattern in patterns )

class TrafficLightController :
    """Smart traffic light controller with emergency prioritization (EAST/WEST only)"""
    def __init__ (self ,serial_port =None ):

        self .state ={'east':'red','west':'red'}
        self .normal_cycle =['east','west']
        self .current_phase =0 
        self .green_duration =10 
        self .yellow_duration =3 
        self .timer =self .green_duration 
        self .phase_state ='green'
        self .emergency_override =False 
        self .emergency_direction =None 
        self .serial_port =serial_port 
        self .state_changed_callback =None 

    def update_cycle (self ):
        """Update traffic light cycle"""
        if self .emergency_override :
            return self ._handle_emergency_cycle ()

        return self ._handle_normal_cycle ()

    def _handle_normal_cycle (self ):
        """Handle normal traffic cycle"""
        active_direction =self .normal_cycle [self .current_phase ]
        self .timer -=1 

        if self .timer <0 :
            if self .phase_state =='green':
                self .phase_state ='yellow'
                self .timer =self .yellow_duration 
            else :
                self .phase_state ='green'
                self .current_phase =(self .current_phase +1 )%len (self .normal_cycle )
                self .timer =self .green_duration 


        for direction in self .state :
            if direction ==active_direction :
                self .state [direction ]=self .phase_state 
            else :
                self .state [direction ]='red'

        self ._send_serial_signal (active_direction ,self .phase_state )
        return self .state 

    def _handle_emergency_cycle (self ):
        """Handle emergency vehicle prioritization (EAST/WEST only)"""
        if self .emergency_direction in self .state :
            for direction in self .state :
                self .state [direction ]='green'if direction ==self .emergency_direction else 'red'
            self ._send_serial_signal (self .emergency_direction ,'green')
        return self .state 

    def trigger_emergency (self ,direction ):
        """Trigger emergency vehicle priority"""
        self .emergency_override =True 
        self .emergency_direction =direction 
        print (f"EMERGENCY: Prioritizing {direction .upper ()} direction")
        return self .state 

    def clear_emergency (self ):
        """Clear emergency override"""
        self .emergency_override =False 
        self .emergency_direction =None 
        print ("Emergency override cleared - resuming normal cycle")
        return self .state 

    def _send_serial_signal (self ,direction ,state ):
        """Send serial signals for traffic light changes"""
        if state =='green'and self .serial_port :
            signal_map ={
            'east':'EG',
            'west':'WG'
            }

            signal =signal_map .get (direction )
            if signal :
                try :
                    self .serial_port .write (f"{signal }\n".encode ())
                    print (f"Sent serial signal: {signal }")
                except Exception as e :
                    print (f"Serial send error: {e }")

class SerialManager :
    """Manages serial communication with auto-detection"""
    def __init__ (self ):
        self .serial_port =None 
        self .connect ()

    def connect (self ):
        """Auto-detect and connect to serial port"""
        ports_to_try =[]

        if platform .system ()=="Darwin":
            ports_to_try =[f"/dev/tty.usbserial-{i }"for i in range (10 )]+["/dev/tty.SLAB_USBtoUART"]
        elif platform .system ()=="Windows":
            ports_to_try =[f"COM{i }"for i in range (1 ,20 )]
        else :
            ports_to_try =[f"/dev/ttyUSB{i }"for i in range (10 )]+[f"/dev/ttyACM{i }"for i in range (10 )]

        for port in ports_to_try :
            try :
                self .serial_port =serial .Serial (port ,115200 ,timeout =1 )
                print (f"Serial connected to {port }")
                return 
            except :
                continue 

        print ("No serial port found")





class SerialReaderThread (QThread ):
    """Background serial reader emitting lines to UI"""
    line_received =pyqtSignal (str )
    status_changed =pyqtSignal (str )
    error_occurred =pyqtSignal (str )

    def __init__ (self ,port_name :str ,baud_rate :int =115200 ,parent =None ):
        super ().__init__ (parent )
        self .port_name =port_name 
        self .baud_rate =baud_rate 
        self ._running =False 
        self ._ser =None 

    def run (self ):
        try :
            self ._ser =serial .Serial (self .port_name ,self .baud_rate ,timeout =1 )
            self .status_changed .emit (f"Connected {self .port_name } @ {self .baud_rate }")
            self ._running =True 
            while self ._running :
                try :
                    raw =self ._ser .readline ()
                    if not raw :
                        continue 
                    try :
                        line =raw .decode (errors ='ignore').strip ()
                    except Exception :
                        line =str (raw )
                    if line :
                        self .line_received .emit (line )
                except Exception as e :
                    self .error_occurred .emit (str (e ))
                    break 
        except Exception as e :
            self .error_occurred .emit (f"Serial open failed: {e }")
        finally :
            try :
                if self ._ser and self ._ser .is_open :
                    self ._ser .close ()
            except Exception :
                pass 
            self .status_changed .emit ("Disconnected")

    def stop (self ):
        self ._running =False 
        try :
            if self ._ser and self ._ser .is_open :
                self ._ser .close ()
        except Exception :
            pass 

class SmartTrafficController :
    """Smart traffic controller with automatic emergency vehicle detection"""

    def __init__ (self ):
        self .current_state ={
        'north':'red',
        'east':'red',
        'south':'red',
        'west':'red'
        }
        self .emergency_active =False 
        self .emergency_direction =None 
        self .emergency_start_time =None 
        self .normal_cycle =['north','east','south','west']
        self .current_phase =0 
        self .phase_timer =10 
        self .emergency_duration =30 

    def trigger_emergency (self ,direction ,plate_number ):
        """Auto-trigger emergency for detected emergency vehicle"""
        direction_map ={
        0 :'north',
        1 :'east',
        2 :'south',
        3 :'west'
        }

        emergency_dir =direction_map .get (direction ,'north')

        if not self .emergency_active :
            self .emergency_active =True 
            self .emergency_direction =emergency_dir 
            self .emergency_start_time =time .time ()


            for direction in self .current_state :
                if direction ==emergency_dir :
                    self .current_state [direction ]='green'
                else :
                    self .current_state [direction ]='red'

            print (f"🚨 EMERGENCY TRIGGERED: {emergency_dir .upper ()} direction for {plate_number }")


            self ._log_emergency_event (emergency_dir ,plate_number )

    def update_cycle (self ):
        """Update traffic light cycle"""
        current_time =time .time ()


        if self .emergency_active :
            if current_time -self .emergency_start_time >self .emergency_duration :
                self .clear_emergency ()
            return self .current_state 



        return self .current_state 

    def clear_emergency (self ):
        """Clear emergency state"""
        self .emergency_active =False 
        self .emergency_direction =None 
        self .emergency_start_time =None 
        print (" Emergency cleared - resuming normal traffic cycle")

    def _log_emergency_event (self ,direction ,plate_number ):
        """Log emergency events to file"""
        try :
            timestamp =datetime .now ().strftime ("%Y-%m-%d %H:%M:%S")
            with open ('emergency_log.txt','a')as f :
                f .write (f"{timestamp } - EMERGENCY: {direction .upper ()} direction triggered by {plate_number }\n")
        except Exception as e :
            print (f"❌ Error logging emergency: {e }")





class HighPerformanceCamera (QThread ):
    """Ultra-accurate camera processing with enhanced detection"""

    frame_ready =pyqtSignal (int ,np .ndarray ,dict )
    plate_detected =pyqtSignal (int ,str ,dict )
    violation_alert =pyqtSignal (dict )

    def __init__ (self ,camera_id ,mode ="PRODUCTION"):
        super ().__init__ ()
        self .camera_id =camera_id 
        self .mode =mode 
        self .running =False 
        self .cap =None 


        self .frame_count =0 
        self .fps_start =time .time ()
        self .fps_counter =0 
        self .current_fps =0 


        self .plate_detector =UltraAccuratePlateDetector ()
        self .ocr_processor =AdaptiveOCRProcessor ()


        self .vehicle_detector =VehicleDetector ()


        self .traffic_controller =SmartTrafficController ()


        self .road_divider_manager =RoadDividerManager (Config .CAMERA_WIDTH ,Config .CAMERA_HEIGHT )


        self .frame_skip_counter =0 
        self .last_frame_time =time .time ()
        self .stable_detections =[]

    def set_mode (self ,mode ):
        """Switch between TEST and PRODUCTION modes"""
        self .mode =mode 

    def run (self ):
        """Main camera processing loop"""
        self .running =True 


        self .cap =cv2 .VideoCapture (self .camera_id )

        if self .cap .isOpened ():

            self .cap .set (cv2 .CAP_PROP_FRAME_WIDTH ,Config .CAMERA_WIDTH )
            self .cap .set (cv2 .CAP_PROP_FRAME_HEIGHT ,Config .CAMERA_HEIGHT )
            self .cap .set (cv2 .CAP_PROP_FPS ,Config .TARGET_FPS )
            self .cap .set (cv2 .CAP_PROP_BUFFERSIZE ,1 )
            print (f"✅ Camera {self .camera_id } initialized with enhanced settings")
        else :
            print (f"❌ Camera {self .camera_id } failed to open")
            return 

        while self .running :
            start_time =time .time ()

            ret ,frame =self .cap .read ()
            if not ret :
                continue 

            self .frame_count +=1 
            self .frame_skip_counter +=1 


            frame_data =self ._process_frame_adaptive (frame )


            self .frame_ready .emit (self .camera_id ,frame ,frame_data )


            self ._update_fps ()


            elapsed =time .time ()-start_time 
            target_frame_time =1.0 /Config .TARGET_FPS 
            if elapsed <target_frame_time :
                time .sleep (target_frame_time -elapsed )

    def _process_frame_adaptive (self ,frame ):
        """ADAPTIVE frame processing with smart performance optimization"""
        frame_data ={
        'plates':[],
        'detected_texts':[],
        'fps':self .current_fps ,
        'mode':self .mode ,
        'processing':False ,
        'accuracy_score':0.0 ,
        'performance_mode':'NORMAL'
        }


        performance_mode =self .ocr_processor .performance_monitor .update_fps (self .current_fps )
        frame_data ['performance_mode']=performance_mode 


        if self .frame_skip_counter >=Config .PROCESS_EVERY_N_FRAMES :
            self .frame_skip_counter =0 
            frame_data ['processing']=True 


            detected_plates =self .plate_detector .detect_plates_ultra_accurate (frame )
            frame_data ['plates']=detected_plates 


            if detected_plates :
                quality_scores =[p ['quality_score']for p in detected_plates ]
                frame_data ['accuracy_score']=sum (quality_scores )/len (quality_scores )


            for plate in detected_plates :
                self .ocr_processor .process_plate_async (
                plate ,self .frame_count ,self .camera_id 
                )


            if Config .RASH_DRIVING_ENABLED and Config .VEHICLE_DETECTION_ENABLED :
                violations =[]
                timestamp =time .time ()


                vehicle_dets =self .vehicle_detector .detect (frame )


                center_x =None 
                if self .road_divider_manager .divider_lines :
                    for div in self .road_divider_manager .divider_lines :
                        if div ['type']=='center_divider':
                            center_x =div ['x']
                            break 

                if center_x is not None :
                    for (x1 ,y1 ,x2 ,y2 ,conf ,cls )in vehicle_dets :

                        if (x1 -Config .LANE_VIOLATION_BUFFER )<=center_x <=(x2 +Config .LANE_VIOLATION_BUFFER ):
                            violation ={
                            'type':'CENTER_DIVIDER_CROSSING',
                            'severity':'HIGH',
                            'description':'Vehicle crossed center divider',
                            'position':(x1 ,y1 ,x2 ,y2 ),
                            'timestamp':timestamp ,
                            'vehicle_id':f"c_{x1 }_{y1 }_{int (conf *100 )}",
                            'cls':int (cls ),
                            'conf':float (conf )
                            }
                            violations .append (violation )
                            frame_data .setdefault ('violations',[]).append (violation )

                if violations :
                    frame_data ['lane_violations']=len (violations )
                    frame_data ['violation_stats']=self .road_divider_manager .get_violation_stats ()

                frame_data ['vehicle_detections']=vehicle_dets


        ocr_results =self .ocr_processor .get_ocr_results ()


        indian_results =[]
        for result in ocr_results :
            if (result ['camera_id']==self .camera_id and 
            result .get ('validation_score',0 )>0.8 ):
                indian_results .append (result )


                if result .get ('is_emergency',False ):
                    self .traffic_controller .trigger_emergency (
                    self .camera_id ,
                    result ['text']
                    )


        current_time =time .time ()
        if current_time -self .last_frame_time >0.5 :
            self .stable_detections =indian_results 
            self .last_frame_time =current_time 

        frame_data ['detected_texts']=self .stable_detections 


        for result in indian_results :
            if result .get ('final_confidence',0 )>Config .CONFIDENCE_BOOST_THRESHOLD :
                self .plate_detected .emit (
                self .camera_id ,
                result ['text'],
                result 
                )


        if frame_data .get ('violations'):
            for violation in frame_data ['violations']:

                violation ['camera_id']=self .camera_id 
                violation ['timestamp_formatted']=datetime .now ().strftime ("%H:%M:%S")

        return frame_data 

    def _update_fps (self ):
        """Update FPS counter"""
        self .fps_counter +=1 
        if time .time ()-self .fps_start >=1.0 :
            self .current_fps =self .fps_counter 
            self .fps_counter =0 
            self .fps_start =time .time ()

    def stop (self ):
        """Stop camera processing"""
        self .running =False 
        if self .cap :
            self .cap .release ()
        self .ocr_processor .stop ()

        indian_patterns =[

        r'^[A-Z]{2}[0-9]{2}[A-Z]{1}[0-9]{4}$',
        r'^(AMB|POL|FIRE|ARMY|NAVY|AF)[0-9A-Z]{4,8}$',
        ]

        for pattern in indian_patterns :
            if re .match (pattern ,text ):
                return 1.0 

        return 0.0

    def export_violation_report (self ):
        """Export detailed violation report"""
        try :
            from datetime import datetime 
            import json 


            all_violations =[]
            for camera in self .cameras :
                if hasattr (camera ,'road_divider_manager'):
                    for track in camera .road_divider_manager .vehicle_tracks .values ():
                        all_violations .extend (track ['violations'])


            report ={
            'generated_at':datetime .now ().isoformat (),
            'total_violations':len (all_violations ),
            'violations':all_violations ,
            'statistics':self .get_violation_summary ()
            }


            filename =f"violation_report_{datetime .now ().strftime ('%Y%m%d_%H%M%S')}.json"
            with open (filename ,'w')as f :
                json .dump (report ,f ,indent =2 )

            print (f"Violation report exported: {filename }")

        except Exception as e :
            print (f"Error exporting violation report: {e }")

    def get_violation_summary (self ):
        """Get summary of violation statistics"""
        summary ={
        'total_violations':0 ,
        'violation_types':{},
        'active_tracks':0 
        }

        for camera in self .cameras :
            if hasattr (camera ,'road_divider_manager'):
                stats =camera .road_divider_manager .get_violation_stats ()
                summary ['total_violations']+=stats ['total_violations']
                summary ['active_tracks']+=stats ['active_tracks']

                for vtype ,count in stats .get ('violation_types',{}).items ():
                    summary ['violation_types'][vtype ]=summary ['violation_types'].get (vtype ,0 )+count 

        return summary

    def switch_mode (self ,mode ):

        self .mode =mode 


        for camera in self .cameras :
            camera .set_mode (mode )

        print (f"🔄 Switched to {mode } mode")

    def start_camera_system (self ):
        """Initialize all cameras with enhanced processing"""
        for i in range (2 ):
            camera =HighPerformanceCamera (i ,self .mode )
            camera .frame_ready .connect (self .on_frame_ready )
            camera .plate_detected .connect (self .on_plate_detected )
            camera .violation_alert .connect (self .show_violation_alert )
            camera .start ()
            self .cameras .append (camera )

        print (" All camera systems started with ultra-accurate detection")

    def on_frame_ready (self ,camera_id ,frame ,data ):
        """Handle new camera frame with sophisticated processing"""
        try :

            display_frame =self .draw_sophisticated_overlays (frame .copy (),data )


            height ,width ,channel =display_frame .shape 
            bytes_per_line =3 *width 
            q_image =QImage (display_frame .data ,width ,height ,bytes_per_line ,QImage .Format_RGB888 ).rgbSwapped ()
            pixmap =QPixmap .fromImage (q_image )


            scaled =pixmap .scaled (
            Config .DISPLAY_WIDTH ,
            Config .DISPLAY_HEIGHT ,
            Qt .KeepAspectRatio ,
            Qt .SmoothTransformation 
            )


            if camera_id <len (self .camera_displays ):
                self .camera_displays [camera_id ].setPixmap (scaled )


            self .update_camera_stats (camera_id ,data )


            if data .get ('violations'):
                for violation in data ['violations']:
                    self .add_violation_to_table (violation ,camera_id )

        except Exception as e :
            print (f"Display error camera {camera_id }: {e }")

    def draw_sophisticated_overlays (self ,frame ,data ):
        """Draw sophisticated detection overlays"""


        if hasattr (self ,'cameras')and len (self .cameras )>0 :

            camera =self .cameras [0 ]if self .cameras else None 
            if camera and hasattr (camera ,'road_divider_manager'):
                frame =camera .road_divider_manager .draw_road_system (frame )

        if data .get ('vehicle_detections'):
            for det in data ['vehicle_detections']:
                x1 ,y1 ,x2 ,y2 ,conf ,cls =det 
                
                vehicle_colors ={
                2 :(0 ,255 ,0 ),   # car
                3 :(255 ,0 ,255 ), # motorcycle  
                5 :(0 ,255 ,255 ), # bus
                7 :(255 ,255 ,0 )  # truck
                }
                
                color =vehicle_colors .get (int (cls ),(128 ,128 ,128 ))
                
                cv2 .rectangle (frame ,(x1 ,y1 ),(x2 ,y2 ),color ,2 )
                cv2 .rectangle (frame ,(x1 ,y1 -25 ),(x2 ,y1 ),color ,-1 )
                
                vehicle_type =['','','car','motorcycle','','bus','','truck'][int (cls )]if int (cls )<8 else 'vehicle'
                label =f"{vehicle_type .upper ()} {conf :.2f}"
                cv2 .putText (frame ,label ,(x1 +2 ,y1 -8 ),
                cv2 .FONT_HERSHEY_SIMPLEX ,0.5 ,(0 ,0 ,0 ),1 )

        if data .get ('violations'):
            for violation in data ['violations']:
                if violation .get ('type')=='CENTER_DIVIDER_CROSSING'and 'position'in violation :
                    x1 ,y1 ,x2 ,y2 =violation ['position']

                    color =(0 ,0 ,128 )
                    cv2 .rectangle (frame ,(x1 -8 ,y1 -8 ),(x2 +8 ,y2 +8 ),color ,6 )

                    cv2 .rectangle (frame ,(x1 -8 ,max (0 ,y1 -40 )),(x2 +8 ,y1 -8 ),color ,-1 )
                    cv2 .putText (frame ,"RASH LANE CHANGING",(x1 ,max (15 ,y1 -16 )),
                    cv2 .FONT_HERSHEY_SIMPLEX ,0.7 ,(255 ,255 ,255 ),2 )


        for plate in data .get ('plates',[]):
            x1 ,y1 ,x2 ,y2 =plate ['bbox']
            confidence =plate ['confidence']
            method =plate .get ('method','unknown')


            method_colors ={
            'yolo_primary':(0 ,255 ,255 ),
            'yolo_backup':(255 ,255 ,0 ),
            'opencv_edge':(255 ,165 ,0 ),
            'opencv_morph':(255 ,0 ,255 ),
            'contour':(0 ,255 ,0 )
            }

            color =method_colors .get (method ,(128 ,128 ,128 ))


            cv2 .rectangle (frame ,(x1 ,y1 ),(x2 ,y2 ),color ,2 )
            cv2 .rectangle (frame ,(x1 ,y1 -25 ),(x2 ,y1 ),color ,-1 )


            label =f"{method .upper ()} {confidence :.2f}"
            cv2 .putText (frame ,label ,(x1 +2 ,y1 -8 ),
            cv2 .FONT_HERSHEY_SIMPLEX ,0.5 ,(0 ,0 ,0 ),1 )


        for text_result in data .get ('detected_texts',[]):
            x1 ,y1 ,x2 ,y2 =text_result ['bbox']
            text =text_result ['text']
            is_emergency =text_result ['is_emergency']
            validation_score =text_result .get ('validation_score',0.0 )


            if is_emergency :
                color =(0 ,0 ,255 )
                thickness =4 
            elif validation_score >0.8 :
                color =(0 ,255 ,0 )
                thickness =3 
            else :
                color =(255 ,255 ,255 )
                thickness =2 


            cv2 .rectangle (frame ,(x1 ,y1 ),(x2 ,y2 ),color ,thickness )
            cv2 .putText (frame ,text ,(x1 ,y1 -35 ),
            cv2 .FONT_HERSHEY_SIMPLEX ,0.8 ,color ,2 )


            score_text =f"Q:{validation_score :.1f}"
            cv2 .putText (frame ,score_text ,(x1 ,y2 +20 ),
            cv2 .FONT_HERSHEY_SIMPLEX ,0.5 ,color ,1 )

            if is_emergency :
                cv2 .putText (frame ,"EMERGENCY",(x1 ,y2 +40 ),
                cv2 .FONT_HERSHEY_SIMPLEX ,0.7 ,(0 ,0 ,255 ),2 )


        mode_color =(255 ,165 ,0 )if data .get ('mode')=='TEST'else (0 ,128 ,255 )
        cv2 .putText (frame ,f"MODE: {data .get ('mode','UNKNOWN')}",
        (10 ,30 ),cv2 .FONT_HERSHEY_SIMPLEX ,0.8 ,mode_color ,2 )

        accuracy =data .get ('accuracy_score',0.0 )
        acc_color =(0 ,255 ,0 )if accuracy >0.7 else (255 ,255 ,0 )if accuracy >0.4 else (255 ,0 ,0 )
        cv2 .putText (frame ,f"ACCURACY: {accuracy :.1%}",
        (10 ,60 ),cv2 .FONT_HERSHEY_SIMPLEX ,0.7 ,acc_color ,2 )

        return frame 

    def update_camera_stats (self ,camera_id ,data ):
        """Update ADAPTIVE camera statistics"""
        if camera_id <len (self .stats_displays ):
            stats =self .stats_displays [camera_id ]


            fps =data .get ('fps',0 )
            stats ['fps'].setText (f"{fps :.1f}")


            plates_count =len (data .get ('plates',[]))
            stats ['plates'].setText (str (plates_count ))


            performance_mode =data .get ('performance_mode','NORMAL')
            stats ['performance'].setText (performance_mode )


            accuracy_score =data .get ('accuracy_score',0.0 )
            stats ['accuracy'].setText (f"{accuracy_score :.1%}")
            self .accuracy_scores [camera_id ]=accuracy_score 


            emergency_count =len ([t for t in data .get ('detected_texts',[])if t .get ('is_emergency',False )])
            stats ['emergency'].setText (str (emergency_count ))


            if 'violations'in stats :
                violation_count =len (data .get ('violations',[]))
                stats ['violations'].setText (str (violation_count ))


                if 'violation_stats'in stats and data .get ('violation_stats'):
                    vstats =data ['violation_stats']
                    total_violations =vstats .get ('total_violations',0 )
                    stats ['violation_stats'].setText (f"Total: {total_violations }")

    def on_plate_detected (self ,camera_id ,plate_text ,plate_data ):
        """Handle sophisticated plate detection"""
        self .total_plates_detected +=1 


        row =self .detection_table .rowCount ()
        self .detection_table .insertRow (row )

        timestamp =datetime .now ().strftime ("%H:%M:%S.%f")[:-3 ]
        camera_name =f"CAM-{camera_id +1 }"
        confidence =f"{plate_data .get ('combined_confidence',0.0 ):.1%}"
        method =plate_data .get ('method','unknown').upper ()
        plate_type =" EMERGENCY"if plate_data .get ('is_emergency',False )else "CIVILIAN"


        items =[
        QTableWidgetItem (timestamp ),
        QTableWidgetItem (camera_name ),
        QTableWidgetItem (plate_text ),
        QTableWidgetItem (confidence ),
        QTableWidgetItem (method ),
        QTableWidgetItem (plate_type )
        ]

        for col, item in enumerate(items):
            self.detection_table.setItem(row, col, item)

        if plate_data.get('is_emergency', False):
            self.emergency_count += 1

            self.add_recent_car(
                camera_id, "CAR", plate_text,
                plate_data.get('combined_confidence', 0.0),
                plate_data.get('is_emergency', False)
            )
            for col in range(6):
                item = self.detection_table.item(row, col)
                if item:
                    item.setBackground(QColor("#FFE6E6"))

            self.update_emergency_status(camera_id, plate_text)
        else:
            self.add_recent_car(
                camera_id, "CAR", plate_text,
                plate_data.get('combined_confidence', 0.0),
                plate_data.get('is_emergency', False)
            )

        self.detection_table.scrollToBottom()
        if row > 200:
            self.detection_table.removeRow(0)
